# 添加自定义软件
## 环境准备

```bash
cd source
git checkout v23.05.3
make distclean

# 提前下载
./scripts/feeds update -a
./scripts/feeds install -a

# 配置信息
make menuconfig

# 获取编译工具链
make toolchain/install

# 把使用的编译工具添加到环境变量
export PATH=/home/buildbot/source/staging_dir/host/bin:$PATH
```

## 软件包
OpenWrt 构建系统中的每个软件包都由一个软件包清单文件描述。清单文件负责描述软件包及其功能，并且至少必须提供获取源代码、构建方式以及最终可安装软件包应包含哪些文件的说明。软件包清单还可以包含可选配置脚本的选项、软件包之间的依赖关系等等
```make
include $(TOPDIR)/rules.mk
 
# Name, version and release number
# The name and version of your package are used to define the variable to point to the build directory of your package: $(PKG_BUILD_DIR)
PKG_NAME:=helloworld
PKG_VERSION:=1.0
PKG_RELEASE:=1
 
# Source settings (i.e. where to find the source codes)
# This is a custom variable, used below
SOURCE_DIR:=/home/buildbot/helloworld
 
include $(INCLUDE_DIR)/package.mk
 
# Package definition; instructs on how and where our package will appear in the overall configuration menu ('make menuconfig')
define Package/helloworld
  SECTION:=examples
  CATEGORY:=Examples
  TITLE:=Hello, World!
endef
 
# Package description; a more verbose description on what our package does
define Package/helloworld/description
  A simple "Hello, world!" -application.
endef
 
# Package preparation instructions; create the build directory and copy the source code. 
# The last command is necessary to ensure our preparation instructions remain compatible with the patching system.
define Build/Prepare
		mkdir -p $(PKG_BUILD_DIR)
		cp $(SOURCE_DIR)/* $(PKG_BUILD_DIR)
		$(Build/Patch)
endef
 
# Package build instructions; invoke the target-specific compiler to first compile the source file, and then to link the file into the final executable
define Build/Compile
		$(TARGET_CC) $(TARGET_CFLAGS) -o $(PKG_BUILD_DIR)/helloworld.o -c $(PKG_BUILD_DIR)/helloworld.c
		$(TARGET_CC) $(TARGET_LDFLAGS) -o $(PKG_BUILD_DIR)/$1 $(PKG_BUILD_DIR)/helloworld.o
endef
 
# Package install instructions; create a directory inside the package to hold our executable, and then copy the executable we built previously into the folder
define Package/helloworld/install
		$(INSTALL_DIR) $(1)/usr/bin
		$(INSTALL_BIN) $(PKG_BUILD_DIR)/helloworld $(1)/usr/bin
endef
 
# This command is always the last, it uses the definitions and variables we give above in order to get the job done
$(eval $(call BuildPackage,helloworld))
```
包含大量变量和智能默认值，因此通常无需像上面那样提供明确的指令
## 关联包
```bash
# 回到openwrt的源码文件夹
cd openwrt
touch feeds.conf
```
在这个文件里面添加我们自己的软件包
```
src-link mypackages /home/jiao/mypackages
```

下面可以进行编译安装
```bash
./scripts/feeds update mypackages
./scripts/feeds install -a -p mypackages
```
## 实际构建
在实际构建一个操作系统的时候, 我们需要在config里面选中这个文件
```bash
make menuconfig
```
在实际的设置界面里面, 可以在`example-->helloworld`找到实际的这个软件的配置
之后可以使用命令`make package/helloworld/compile`进行编译
如果一切顺利，我们将在 `bin/packages/<arch>/mypackages` 文件夹中看到一个名为 `helloworld_1.0-1_<arch>.ipk` 的全新软件包

## 复杂编译

在一个复杂的项目进行编译的时候, 可以把打包和实际的代码文件分开处理
创建一个单独的代码文件夹, 里面存放Makefile文件以及代码
### 代码编译文件
```make
# 默认目标：编译生成 helloworld 可执行文件
all: helloworld

# 编译器、编译参数、链接参数（会被 OpenWrt 构建系统覆盖）
CC = gcc
CFLAGS = -Wall
LDFLAGS =

# 自动匹配所有头文件、C 源码文件
DEPS = $(wildcard *.h)
SRC = $(wildcard *.c)
OBJ = $(patsubst %.c,%.o,$(SRC))

# 编译规则：.c 文件生成 .o 目标文件
%.o: %.c $(DEPS)
	$(CC) -c -o $@ $< $(CFLAGS)

# 链接规则：所有 .o 文件生成最终可执行文件
helloworld: $(OBJ)
	$(CC) -o $@ $^ $(LDFLAGS)

# 清理编译产物
.PHONY: clean
clean:
	rm -f helloworld *.o
```
### 包内Makefile
```bash
include $(TOPDIR)/rules.mk
 
# Name, version and release number
# The name and version of your package are used to define the variable to point to the build directory of your package: $(PKG_BUILD_DIR)
PKG_NAME:=helloworld
PKG_VERSION:=1.0
PKG_RELEASE:=1
 
# Source settings (i.e. where to find the source codes)
# This is a custom variable, used below
SOURCE_DIR:=/home/jiao/helloworld
 
include $(INCLUDE_DIR)/package.mk
 
# Package definition; instructs on how and where our package will appear in the overall configuration menu ('make menuconfig')
define Package/helloworld
  SECTION:=examples
  CATEGORY:=Examples
  TITLE:=Hello, World!
endef
 
# Package description; a more verbose description on what our package does
define Package/helloworld/description
  A simple "Hello, world!" -application.
endef
 
# Package preparation instructions; create the build directory and copy the source code. 
# The last command is necessary to ensure our preparation instructions remain compatible with the patching system.
define Build/Prepare
		mkdir -p $(PKG_BUILD_DIR)
		cp $(SOURCE_DIR)/* $(PKG_BUILD_DIR)
		$(Build/Patch)
endef
 
# Package build instructions; invoke the target-specific compiler to first compile the source file, and then to link the file into the final executable
# 这个位置是实际的编译的位置
define Build/Compile
        $(MAKE) -C $(PKG_BUILD_DIR) \
               CC="$(TARGET_CC)" \
           CFLAGS="$(TARGET_CFLAGS)" \
          LDFLAGS="$(TARGET_LDFLAGS)"
endef
 
# Package install instructions; create a directory inside the package to hold our executable, and then copy the executable we built previously into the folder
define Package/helloworld/install
		$(INSTALL_DIR) $(1)/usr/bin
		$(INSTALL_BIN) $(PKG_BUILD_DIR)/helloworld $(1)/usr/bin
endef
 
# This command is always the last, it uses the definitions and variables we give above in order to get the job done
$(eval $(call BuildPackage,helloworld))
```
