# 内核线程

设备驱动中，如果需要几个并发执行的任务，可以启动内核线程

## 常见API

```c
#define kthread_run(threadfn, data, namefmt, ...)			   \
({									   \
	struct task_struct *__k						   \
		= kthread_create(threadfn, data, namefmt, ## __VA_ARGS__); \
	if (!IS_ERR(__k))						   \
		wake_up_process(__k);					   \
	__k;								   \
})

#define kthread_create(threadfn, data, namefmt, arg...) \
	kthread_create_on_node(threadfn, data, NUMA_NO_NODE, namefmt, ##arg)

struct task_struct *kthread_create_on_node(int (*threadfn)(void *data),
					   void *data, int node,
					   const char namefmt[],
					   ...)
{
	struct task_struct *task;
	va_list args;

	va_start(args, namefmt);
	task = __kthread_create_on_node(threadfn, data, node, namefmt, args);
	va_end(args);

	return task;
}
__kthread_create_on_node // 把信息记录在kthread_create_list里面
	=> kthreadd // 实际处理创建任务, 这个是在rest_init里面创建的, 
    	// 紧跟着init任务, pid是2
		=> create_kthread // 创建, 失败的时候通知一下完成量
    		=> kernel_thread
```

### 后台处理线程

- `kthreadd`：内核里的“线程工厂/代管者”
- `kthread_run()`：把创建请求丢给 `kthreadd`
- 新创建出来的线程：才去执行各自的 `threadfn`

```bash
PID   USER     TIME  COMMAND
    1 0         0:04 init
    2 0         0:00 [kthreadd]
```

```c
// 创建线程使用的数据
struct kthread_create_info
{
	/* Information passed to kthread() from kthreadd. */
	int (*threadfn)(void *data); // 线程函数
	void *data; // 线程函数的参数
    // NUMA 节点可以理解成：多路/多芯片 CPU 机器里，内存和 CPU 组成的集合
    // 通常包含一组 CPU 核，以及离这些 CPU 更近的一部分内存。
	// CPU 访问“本节点内存”更快，访问“别的节点内存”更慢。
	int node; // 线程绑定的 NUMA 节点

	/* Result passed back to kthread_create() from kthreadd. */
	struct task_struct *result; // 最后创建的线程的 task_struct 结构体指针
	struct completion *done; // 完成结构体指针，用于同步创建线程的过程

	struct list_head list;
};
```



```c
int kthreadd(void *unused)
{
	struct task_struct *tsk = current;

	/* Setup a clean context for our children to inherit. */
	set_task_comm(tsk, "kthreadd"); // 配置当前线程的名字为 kthreadd
	ignore_signals(tsk);
	set_cpus_allowed_ptr(tsk, housekeeping_cpumask(HK_FLAG_KTHREAD));
	set_mems_allowed(node_states[N_MEMORY]);

	current->flags |= PF_NOFREEZE;
	cgroup_init_kthreadd(); // 初始化 kthreadd 的 cgroup

	for (;;) {
		set_current_state(TASK_INTERRUPTIBLE);
		if (list_empty(&kthread_create_list))
			schedule();
		__set_current_state(TASK_RUNNING);

		spin_lock(&kthread_create_lock);
		while (!list_empty(&kthread_create_list)) {
			struct kthread_create_info *create;

			create = list_entry(kthread_create_list.next,
					    struct kthread_create_info, list);
			list_del_init(&create->list);
			spin_unlock(&kthread_create_lock);
			// 创建内核线程
			create_kthread(create);

			spin_lock(&kthread_create_lock);
		}
		spin_unlock(&kthread_create_lock);
	}

	return 0;
}
```

## 创建新的线程

kernel\fork.c

```c
pid_t kernel_thread(int (*fn)(void *), void *arg, unsigned long flags)
{
	struct kernel_clone_args args = {
		// 加上 CLONE_VM：让这个内核线程和当前任务共享地址空间
		// & ~CSIGNAL：把退出信号位清掉，避免和内核线程场景冲突
		.flags		= ((lower_32_bits(flags) | CLONE_VM |
				    CLONE_UNTRACED) & ~CSIGNAL),
		.exit_signal	= (lower_32_bits(flags) & CSIGNAL),
		.stack		= (unsigned long)fn, // 不是普通意义上的栈，而是内核线程的入口函数
		.stack_size	= (unsigned long)arg,
	};

	return kernel_clone(&args);
}

copy_process
    => dup_task_struct //复制任务的task_struct, 初始化各种字段
    	// （调度、锁、统计、pending signal 等）
    // 这部分是“把新任务从空壳变成可运行实体”的主体
    => copy_creds()（凭据）
	=> copy_files() / copy_fs()（文件表、fs 上下文）
	=> copy_sighand() / copy_signal()（信号处理）
	=> copy_mm()（地址空间）

    	=> dup_mm
    		=> dup_mmap
    			=> copy_page_range // 页表的复制
	=> copy_namespaces()（命名空间）
	=> copy_io()、copy_thread()（架构线程上下文）
// copy_thread 是不同的架构里面使用不同的复制函数
// 给新建的 task 初始化 CPU 上下文和内核栈里的寄存器现场，
// 让它之后能从 ret_from_fork 正常开始运行
// (arm)
int copy_thread(unsigned long clone_flags, unsigned long stack_start,
		unsigned long stk_sz, struct task_struct *p, unsigned long tls)
{
	struct thread_info *thread = task_thread_info(p);
	struct pt_regs *childregs = task_pt_regs(p);
	// 先把保存的寄存器/调用现场清干净
	memset(&thread->cpu_context, 0, sizeof(struct cpu_context_save));

#ifdef CONFIG_CPU_USE_DOMAINS
	/*
	 * Copy the initial value of the domain access control register
	 * from the current thread: thread->addr_limit will have been
	 * copied from the current thread via setup_thread_stack() in
	 * kernel/fork.c
	 */
    // 复制 domain / 地址访问相关状态
	thread->cpu_domain = get_domain();
#endif

	if (likely(!(p->flags & (PF_KTHREAD | PF_IO_WORKER)))) {
        // 不是 PF_KTHREAD | PF_IO_WORKER
        //	直接复制当前进程的 pt_regs
		*childregs = *current_pt_regs();
		childregs->ARM_r0 = 0; // 设置r0为0
		if (stack_start)
			childregs->ARM_sp = stack_start;
	} else {
        // 不复制用户态寄存器, 清空 pt_regs
		memset(childregs, 0, sizeof(struct pt_regs));
		thread->cpu_context.r4 = stk_sz; // 设置参数
		thread->cpu_context.r5 = stack_start; // 设置函数地址
		childregs->ARM_cpsr = SVC_MODE;
	}
    // 设置开始执行的位置
    // 将来被调度上来时，会从 ret_from_fork() 进入
	thread->cpu_context.pc = (unsigned long)ret_from_fork;
	thread->cpu_context.sp = (unsigned long)childregs;

	clear_ptrace_hw_breakpoint(p);

	if (clone_flags & CLONE_SETTLS)
		thread->tp_value[0] = tls;
	thread->tp_value[1] = get_tpuser();

	thread_notify(THREAD_NOTIFY_COPY, thread);

#ifdef CONFIG_STACKPROTECTOR_PER_TASK
	thread->stack_canary = p->stack_canary;
#endif

	return 0;
}
```

处理fork返回的是一个汇编微妙

```assembly
/*
 * This is how we return from a fork.
 */
ENTRY(ret_from_fork)
	bl	schedule_tail
	cmp	r5, #0
	movne	r0, r4
	badrne	lr, 1f
	retne	r5
1:	get_thread_info tsk
	b	ret_slow_syscall
ENDPROC(ret_from_fork)
```
