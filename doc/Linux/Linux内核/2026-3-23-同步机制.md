# 同步机制

## 原子变量

实际是通过不同的内核的原子指令实现的对一个变量的操作,保证这个变量的操作是原子的

#### 原子整型变量

```c
typedef struct {
int counter;
} atomic_t;
```

##### ATOMIC_INIT()

定义整型原子变量并设置初始值

```c
#define ATOMIC_INIT(i) { (i) }
atomic_t data = ATOMIC_INIT(int i)
```

WRITE_ONCE 宏保证这个写操作不会被编译器优化或重排序

##### atomic_set()

用于设置整型原子变量的值

```c
#define __WRITE_ONCE(x, val)                        \
do {                                                \
    *(volatile typeof(x) *)&(x) = (val);            \
} while (0)
#define arch_atomic_set(v, i)			__WRITE_ONCE(((v)->counter), (i))
#define atomic_set(v, i) WRITE_ONCE(((v)->counter), (i))
atomic_set(atomic_t *v,int i)
```

这个操作并不是原子操作, 只是使用volatile保证确实是操作到这个数据

##### atomic_read()

获取原子变量的值

```c
atomic_read(atomic_t *v)
```

##### atomic_add()/atomic_sub()

整型原子变量的加/减

```c
static inline void atomic_add(int i, atomic_t *v) //整型原子变量加i
static inline void atomic_sub(int i, atomic_t *v) //整型原子变量减i
```

## 自旋锁

- ARM64 自旋锁本质是一个 32 位整数，0 表示未加锁，1 表示已加锁。
- 加锁用原子指令循环尝试设置为1，解锁直接原子写0。
- 在等待锁的时候不会调度, 而是进行死锁, 为了避免优先级翻转, 获取自旋锁时，内核会自动禁用当前 CPU 的抢占（preempt_disable ()）；释放锁时，自动启用抢占（preempt_enable ()）

设自旋锁为变量 lock，整数 0 表示锁是空闲状态，整数 pid 表示线程 ID

- CAS(lock, 0, pid) 就表示自旋锁的**加锁**操作
- CAS(lock, pid,0) 则表示自旋锁的**解锁**操作

### 单核加锁

因为单核下不会有多个 CPU 并发执行，所以不需要真正的锁，只要保证本任务不被切换即可, 读写锁的实现也是同样的

include\linux\spinlock.h

```c
static __always_inline void spin_lock(spinlock_t *lock)
{
	raw_spin_lock(&lock->rlock);
}

#define raw_spin_lock(lock)	_raw_spin_lock(lock)
#define _raw_spin_lock(lock)			__LOCK(lock)

// preempt_disable()：关闭内核抢占，防止当前任务被切换出去，保证临界区的原子性
// 在单核（UP）下，自旋锁不会真正加锁，只是通过禁止抢占和静态分析标记，
// 保证临界区不会被打断
#define __LOCK(lock) \
  do { preempt_disable(); ___LOCK(lock); } while (0)
#define ___LOCK(lock) \
  do { __acquire(lock); (void)(lock); } while (0)

# define __acquire(x)	(void)0
```

### 多核

多核实现

include\asm-generic\qrwlock.h

```c
// 加锁时，线程原子性地获取一个“票号”（next++），
// 然后自旋等待 owner==自己票号，解锁时 owner++。
typedef struct {
	union {
		u32 slock; // 自旋锁的原始 32 位值，便于原子操作
		struct __raw_tickets {
#ifdef __ARMEB__
			u16 next;
			u16 owner;
#else
			u16 owner; // 当前持有锁的线程的票号，表示哪个线程正在持有自旋锁
			           // 通过比较 owner 和 next 的值，可以判断是否有线程正在等待获取锁，
					   // 以及哪个线程是下一个将要获取锁的线程。
			u16 next; // 下一个等待获取锁的线程的票号，表示下一个将要获取自旋锁的线程
#endif
		} tickets;
	};
} arch_spinlock_t;

typedef struct {
	u32 lock; // 正数表示读锁的数量，负数表示写锁被持有
} arch_rwlock_t;
```

##### spin_lock

1. 原子性地获取自己的票号（next），并更新锁的 next 字段。
2. 不断检查 owner 字段，只有当 owner==自己票号时，才获得锁。
3. 这样保证了所有线程公平排队，按顺序依次获得锁。

> arm

```c
static inline void arch_spin_lock(arch_spinlock_t *lock)
{
	unsigned long tmp;
	u32 newval;
	// 定义一个局部变量 lockval，用于存储从内存中加载的自旋锁的当前状态。
	// 这个变量是一个 arch_spinlock_t 类型的结构体，包含了自旋锁的原始 
	// 32 位值以及 owner 和 next 的票号信息。
	arch_spinlock_t lockval; 

	prefetchw(&lock->slock);
	__asm__ __volatile__(
"1:	ldrex	%0, [%3]\n" // 从内存地址 %3（即 lock->slock）加载当前的锁值到寄存器 %0 中，并标记该地址为独占访问。
"	add	%1, %0, %4\n" // 将当前锁值 %0 加上一个偏移量 %4（即 1 << TICKET_SHIFT），得到新的锁值，并存储在寄存器 %1 中。
"	strex	%2, %1, [%3]\n" // 尝试将新的锁值 %1 存储回内存地址 %3（即 lock->slock）。如果在此期间有其他处理器修改了该地址，存储操作会失败，并将结果存储在寄存器 %2 中。
"	teq	%2, #0\n" // 检查存储操作是否成功（即 %2 是否为 0）。如果存储失败，说明有其他处理器修改了锁值，当前处理器需要重新尝试获取锁。
"	bne	1b" // 如果存储失败，跳转回标签 1 处，重新执行获取锁的过程。
	: "=&r" (lockval), "=&r" (newval), "=&r" (tmp)
	: "r" (&lock->slock), "I" (1 << TICKET_SHIFT)
	: "cc");

	while (lockval.tickets.next != lockval.tickets.owner) {
		//// 如果当前线程的票号（lockval.tickets.next）不等于持有锁的线程的票号
		// （lockval.tickets.owner），说明当前线程需要等待获取锁，此时调用 WFE 
		// 指令进入低功耗等待状态，直到有其他线程释放锁并发出 SEV 信号，唤醒当前
		// 线程继续尝试获取锁。
		wfe(); 
		// 在被唤醒后，重新加载锁的状态到 lockval 变量中，
		// 以检查当前线程是否已经成功获取了锁。
		lockval.tickets.owner = READ_ONCE(lock->tickets.owner);
	}
	// 获取锁成功后，执行内存屏障，确保在获取锁之前的内存操作对其他处理器可见。
	smp_mb();
}
```

## 读写锁

在实际应用的时候有一个情况是可以**多个进行读取**, 但是**只有一个可以改**

读取的时候等待是否有正在写入, 有的话进行等待, 写的时候等待没有别的读写, 使用一个标志位进行区分

### 读锁

```c
#define read_lock(lock)		_raw_read_lock(lock)
void __lockfunc _raw_read_lock(rwlock_t *lock)
{
	__raw_read_lock(lock);
}

static inline void __raw_read_lock(rwlock_t *lock)
{
	preempt_disable();
	rwlock_acquire_read(&lock->dep_map, 0, 0, _RET_IP_);
	LOCK_CONTENDED(lock, do_raw_read_trylock, do_raw_read_lock);
}

#define LOCK_CONTENDED(_lock, try, lock) \
	lock(_lock)

# define do_raw_read_lock(rwlock)	do {__acquire(lock); arch_read_lock(&(rwlock)->raw_lock); } while (0)

static inline void arch_read_lock(arch_rwlock_t *rw)
{
	unsigned long tmp, tmp2;

	prefetchw(&rw->lock); // 预获取 rw->lock 的地址到缓存中，以提高后续访问的性能。
	__asm__ __volatile__(
"	.syntax unified\n"
"1:	ldrex	%0, [%2]\n" // 原子读取 rw->lock 到 tmp，标记为独占访问
"	adds	%0, %0, #1\n" // tmp 加 1，表示增加一个读者
"	strexpl	%1, %0, [%2]\n" // 如果 lock 不是负数（未被写锁），原子写回新值到 rw->lock，结果存到 tmp2
	WFE("mi") // 如果 lock 是负数（被写锁），进入低功耗等待
"	rsbspl	%0, %1, #0\n" // 如果写回失败或被写锁，回到1重新尝试
"	bmi	1b" // 如果写回失败或被写锁，回到1重新尝试
	: "=&r" (tmp), "=&r" (tmp2)
	: "r" (&rw->lock)
	: "cc");

	smp_mb();
}
```

### 写锁

```c
static inline void arch_write_lock(arch_rwlock_t *rw)
{
	unsigned long tmp;

	prefetchw(&rw->lock);
	__asm__ __volatile__(
"1:	ldrex	%0, [%1]\n" // 原子读取 rw->lock 到 tmp，标记为独占访问
"	teq	%0, #0\n" // 判断锁当前是否为0（无读者/写者）
	WFE("ne") // 如果不为0（有读者/写者）
"	strexeq	%0, %2, [%1]\n" //// 如果为0，尝试原子写入0x80000000（写锁标志），结果存到tmp
"	teq	%0, #0\n" 
"	bne	1b" // 如果写入失败，回到1重新尝试
	: "=&r" (tmp)
	: "r" (&rw->lock), "r" (0x80000000)
	: "cc");

	smp_mb();
}
```

## 顺序锁

自带一个自旋锁, 和一个带锁的序列计数器

> 实际的实现是通过比较是否更改的标记, 进行判断之前是否更改, 是的话用户可以重新读取

### 基础数据

```c
typedef struct {
	/*
	 * Make sure that readers don't starve writers on PREEMPT_RT: use
	 * seqcount_spinlock_t instead of seqcount_t. Check __SEQ_LOCK().
	 */
	seqcount_spinlock_t seqcount;
	spinlock_t lock;
} seqlock_t;
```

### 写操作

```c
static inline void write_seqlock(seqlock_t *sl)
{
    // 使用自旋锁, 保证这个时候只有一个进程调用
    spin_lock(&sl->lock);
    // 递增序列计数器的值（通常加1变为奇数），通知读者有写操作正在进行。
    // 这样读者在检测到序列号为奇数时会重试，保证读到的数据一致性
    do_write_seqcount_begin(&sl->seqcount.seqcount);
}

static inline void do_write_seqcount_begin(seqcount_t *s)
{
	do_write_seqcount_begin_nested(s, 0);
}

static inline void do_write_seqcount_begin_nested(seqcount_t *s, int subclass)
{
	seqcount_acquire(&s->dep_map, subclass, 0, _RET_IP_);
	do_raw_write_seqcount_begin(s);
}

static inline void do_raw_write_seqcount_begin(seqcount_t *s)
{
	kcsan_nestable_atomic_begin();
	s->sequence++; //使序列号变为奇数，通知读者有写操作正在进行
	smp_wmb(); // 写内存屏障，保证写操作的顺序性
}
```

### 读操作

开始获取当前标志

```c
#define raw_read_seqcount_begin(s)					\
({									\
	unsigned _seq = __read_seqcount_begin(s);			\
									\
	smp_rmb();							\
	_seq;								\
})


#define __read_seqcount_begin(s)					\
({									\
	unsigned __seq;							\
									\
	// 是一个单数的时候, 说明在写, 进行等待
	while ((__seq = seqprop_sequence(s)) & 1)			\
		cpu_relax();						\
									\
	kcsan_atomic_next(KCSAN_SEQLOCK_REGION_MAX);			\
	__seq;								\
})
```

其中 read_seqcount_begin 会调用 __read_seqcount_begin，后者会等待 sequence 为偶数（无写者），并返回当前值

读操作结束

```c
static inline unsigned read_seqretry(const seqlock_t *sl, unsigned start)
{
    kcsan_flat_atomic_end();
    return read_seqcount_retry(&sl->seqcount, start);
}

#define read_seqcount_retry(s, start)					\
	do_read_seqcount_retry(seqprop_ptr(s), start)

static inline int do_read_seqcount_retry(const seqcount_t *s, unsigned start)
{
	smp_rmb();
	return do___read_seqcount_retry(s, start);
}

static inline int do___read_seqcount_retry(const seqcount_t *s, unsigned start)
{
	kcsan_atomic_next(0);
	return unlikely(READ_ONCE(s->sequence) != start);
}
```

其中 read_seqcount_retry 会比较当前 sequence 和 start 是否一致，不一致则说明有写操作发生，读者需要重试

```c
void __sched wait_for_completion(struct completion *x)

static inline long __sched
__wait_for_common(struct completion *x,
		  long (*action)(long), long timeout, int state)
{
    // 检查是否可以睡眠
	might_sleep();

	complete_acquire(x); // 未实现
	// 获取自旋锁并且关闭中断
	raw_spin_lock_irq(&x->wait.lock);
	timeout = do_wait_for_common(x, action, timeout, state);
	raw_spin_unlock_irq(&x->wait.lock);

	complete_release(x);

	return timeout;
}

static inline long __sched
do_wait_for_common(struct completion *x,
		   long (*action)(long), long timeout, int state)
{
	if (!x->done) {
		DECLARE_SWAITQUEUE(wait);

		do {
			if (signal_pending_state(state, current)) {
				timeout = -ERESTARTSYS;
				break;
			}
             // 将当前线程加入到等待队列中
			__prepare_to_swait(&x->wait, &wait);
			__set_current_state(state);
			raw_spin_unlock_irq(&x->wait.lock);
			timeout = action(timeout); // 实际的等待
			raw_spin_lock_irq(&x->wait.lock);
		} while (!x->done && timeout);
		__finish_swait(&x->wait, &wait);
		if (!x->done)
			return timeout;
	}
	if (x->done != UINT_MAX)
		x->done--;
	return timeout ?: 1;
}
// kernel\time\timer.c
signed long __sched schedule_timeout(signed long timeout)
{
	struct process_timer timer;
	unsigned long expire;

	switch (timeout)
	{
	case MAX_SCHEDULE_TIMEOUT:
		/*
		 * These two special cases are useful to be comfortable
		 * in the caller. Nothing more. We could take
		 * MAX_SCHEDULE_TIMEOUT from one of the negative value
		 * but I' d like to return a valid offset (>=0) to allow
		 * the caller to do everything it want with the retval.
		 */
		schedule();
		goto out;
	default:
		/*
		 * Another bit of PARANOID. Note that the retval will be
		 * 0 since no piece of kernel is supposed to do a check
		 * for a negative retval of schedule_timeout() (since it
		 * should never happens anyway). You just have the printk()
		 * that will tell you if something is gone wrong and where.
		 */
		if (timeout < 0) {
			printk(KERN_ERR "schedule_timeout: wrong timeout "
				"value %lx\n", timeout);
			dump_stack();
			current->state = TASK_RUNNING;
			goto out;
		}
	}

	expire = timeout + jiffies; // 计算出过期时间

	timer.task = current; // 将当前任务保存到定时器结构中
	// 设置定时器回调函数
	timer_setup_on_stack(&timer.timer, process_timeout, 0);
	// 将定时器添加到定时器链表中，并且设置过期时间
	__mod_timer(&timer.timer, expire, MOD_TIMER_NOTPENDING);
	schedule();
	// 删除单次定时器
	del_singleshot_timer_sync(&timer.timer);

	/* Remove the timer from the object tracker */
	destroy_timer_on_stack(&timer.timer);

	timeout = expire - jiffies;

 out:
	return timeout < 0 ? 0 : timeout;
}
```

### 唤醒

```c
void complete(struct completion *x)
{
	unsigned long flags;
	// 获取锁并保存当前的中断状态
	raw_spin_lock_irqsave(&x->wait.lock, flags);

	if (x->done != UINT_MAX)
		x->done++;
	// 唤醒等待队列中的一个线程
	swake_up_locked(&x->wait);
	// 释放锁并恢复之前的中断状态
	raw_spin_unlock_irqrestore(&x->wait.lock, flags);
}
```
