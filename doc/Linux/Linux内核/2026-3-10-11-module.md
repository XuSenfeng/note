# module

内核里面的模块的加载

## 基础数据

记录在文件include\linux\module.h里面

```c
struct load_info {
	const char *name;
	/* pointer to module in temporary copy, freed at end of load_module() */
	struct module *mod; // 实际的文件中的初始化数据
	Elf_Ehdr *hdr; // 记录elf文件头信息
	unsigned long len; // 文件的长度
	Elf_Shdr *sechdrs; // 节头表, 每个段的目录索引
    // 节头字符表, 记录段的名字, 以及记录字符串表的实际位置
	char *secstrings, *strtab; 
	unsigned long symoffs, stroffs, init_typeoffs, core_typeoffs;
	struct _ddebug *debug;
	unsigned int num_debug;
	bool sig_ok;
#ifdef CONFIG_KALLSYMS
	unsigned long mod_kallsyms_init_off;
#endif
	struct {
		unsigned int sym, str, mod, vers, info, pcpu;
	} index; // 记录各个段的索引, sym符号表(记录字符串表里面的偏移), str(字符串表的索引), mod(.gnu.linkonce.this_module节的索引), vers(版本号), info(记录modinfo段的位置), pcpu
};
```

所有使用MODULE_LICENSE和MODULE_ALIAS等宏定义配置的数据, 会被记录在文件的.modinfo字段里面, 可以使用`get_modinfo`函数进行查找

## 加载

### 前置加载

kernel\kmod.c

```c
int __request_module(bool wait, const char *fmt, ...)
```

在加载的时候使用钩子函数进行安全检查, 所有的钩子函数在`security_hook_heads`, 这个里面只记录链表头

```bash
#define call_int_hook(FUNC, IRC, ...) ({			\
	int RC = IRC;						\
	do {							\
		struct security_hook_list *P;			\
								\
		hlist_for_each_entry(P, &security_hook_heads.FUNC, list) { \
			RC = P->hook.FUNC(__VA_ARGS__);		\
			if (RC != 0)				\
				break;				\
		}						\
	} while (0);						\
	RC;							\
})
```

记录数据在

```c
# 一个联合体, 使用引用头文件的方式, 添加各种回调函数
union security_list_options {
	#define LSM_HOOK(RET, DEFAULT, NAME, ...) RET (*NAME)(__VA_ARGS__);
	#include "lsm_hook_defs.h"
	#undef LSM_HOOK
};
// 记录一个包含很多个链表头的链表
struct security_hook_heads {
	#define LSM_HOOK(RET, DEFAULT, NAME, ...) struct hlist_head NAME;
	#include "lsm_hook_defs.h"
	#undef LSM_HOOK
} __randomize_layout;

/*
 * Security module hook list structure.
 * For use with generic list macros for common operations.
 */
// 实际记录的链表项
struct security_hook_list {
	struct hlist_node		list; // 用于将security_hook_list结构体链接到security_hook_heads中对应hook函数的hlist_head上
	struct hlist_head		*head; // 指向security_hook_heads中对应hook函数的hlist_head
	union security_list_options	hook; // 记录了每个hook函数的地址
	char				*lsm;
} __randomize_layout;

/*
 * Initializing a security_hook_list structure takes
 * up a lot of space in a source file. This macro takes
 * care of the common case and reduces the amount of
 * text involved.
 */
// 初始化一个链表项
#define LSM_HOOK_INIT(HEAD, HOOK) \
	{ .head = &security_hook_heads.HEAD, .hook = { .HEAD = HOOK } }

// 链表的初始化在函数security_add_hooks里面
```

处理一下正在加载的模块的数量, 处理以后使用`/sbin/modprobe`进行实际的加载

### 实际加载

`request_module -> modprobe(用户态) -> finit_module/init_module -> load_module -> do_init_module`

从用户态复制数据

```bash
init_module 
	=> copy_module_from_user 
		=> security_kernel_load_data(处理相关的钩子函数) 
			=> ima_load_data(判断是不是需要IMA认证)
	=> copy_chunked_from_user 从用户空间复制数据到info的hdr
```

在`load_module `函数里面

+ 检查可执行文件的格式合法
+ 是否在黑名单
+ 检查版本
+ 解析布局, 分配内存, 动态加载模块（`insmod`/`finit_module`路径）在加载时会同时准备两块布局：`core_layout`：长期保留 `init_layout`：只给初始化阶段用（`.init*` 等段）
+ 初始化参数
+ 处理重定位
+ 处理sysfs的相关

```bash
load_module
	=> elf_validity_check 查看文件的架构等信息, 记录信息在info里面
	=> setup_load_info 加载多个段的位置, 获取module结构的文件地址
	=> setup_load_info 检查一下elf文件里面的info字段的配置
	=> layout_and_allocate 把模块从ELF镜像变成最终可运行内存布局, 返回mod地址
    	=> check_modinfo 检查各种配置
    	=> layout_sections 给模块的 SHF_ALLOC 各节做内存布局规划，
            把它们分到 core 和 init 两块,统计并更新模块布局边界：text_size、
            ro_size、ro_after_init_size、总 size, 
            init部分的在初始化结束以后会释放
    	=> move_module 分配并且移动数据
    		init部分的地址在mod->init_layout.base
    		core部分的在 mod->core_layout.base
    		按照之前的分配进行拷贝
    => add_unformed_module 把一个“还没完全加载完成”的模块先挂进全局模块
    	列表里，并确保模块名唯一
    => find_module_sections 把模块文件里的各种特殊 section，
    	整理成内核方便直接访问的指针。
    => mod_sysfs_setup 在sysfs文件系统里面创建对应的控制文件
    => do_init_module
```

## 初始化

```c
/**
 * module_init() - driver initialization entry point
 * @x: function to be run at kernel boot time or module insertion
 *
 * module_init() will either be called during do_initcalls() (if
 * builtin) or at module insertion time (if a module).  There can only
 * be one per module.
 */
#define module_init(x)	__initcall(x);
#define __initcall(fn) device_initcall(fn)
#define __define_initcall(fn, id) ___define_initcall(fn, id, .initcall##id)
#define ___define_initcall(fn, id, __sec)			\
	__unique_initcall(fn, id, __sec, __initcall_id(fn))
#define __unique_initcall(fn, id, __sec, __iid)			\
	____define_initcall(fn,					\
		__initcall_stub(fn, __iid, id),			\
		__initcall_name(initcall, __iid, id),		\
		__initcall_section(__sec, __iid))


```

把函数指针（或相对重定位项）放进特定的 `.initcall*` 链接段, 启动时由内核统一遍历这些段并调用, 不同的类型的`initcall`的等级不一样

所有默认加载的模块统一在函数`do_initcalls`函数里面执行

在链接文件里面

```lds
#define INIT_CALLS							\
		__initcall_start = .;					\
		KEEP(*(.initcallearly.init))				\
		INIT_CALLS_LEVEL(0)					\
		INIT_CALLS_LEVEL(1)					\
		INIT_CALLS_LEVEL(2)					\
		INIT_CALLS_LEVEL(3)					\
		INIT_CALLS_LEVEL(4)					\
		INIT_CALLS_LEVEL(5)					\
		INIT_CALLS_LEVEL(rootfs)				\
		INIT_CALLS_LEVEL(6)					\
		INIT_CALLS_LEVEL(7)					\
		__initcall_end = .;
```

执行的时候按照待机依次执行这些函数

1. 你写模块源码（比如 `foo.c`），只写 `module_init` / `module_exit` / `MODULE_LICENSE` 这类宏。
2. 编译时 `modpost` 会生成 `foo.mod.c`。
3. 在这个自动生成文件里，add_header()会写出：
    - `__visible struct module __this_module`
    - 并放进 `.gnu.linkonce.this_module` 节。代码就在`scripts\mod\modpost.c`的add_header函数里面。
4. 最终链接成 `foo.ko` 后，内核加载器在加载时用
    `info->index.mod = find_sec(info, ".gnu.linkonce.this_module");`
    去找到这节，见 module.c:3278。
