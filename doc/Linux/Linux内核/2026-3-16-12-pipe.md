struct pipe_inode_info {

	struct fasync_struct *fasync_readers;
	struct fasync_struct *fasync_writers;
	struct pipe_buffer *bufs; // 管道缓冲区的循环数组，大小由buffers字段指定，每个pipe_buffer结构体表示管道中的一个缓冲区

## 实现

### 创建

pipe() 和 pipe2() 都是 Linux 用户空间用于创建匿名管道的系统调用，但它们有如下区别：

1.  pipe()

-   原型：int pipe(int pipefd[2]);
-   只能创建最基本的管道，不能直接设置额外的标志。
-   创建后，文件描述符的属性（如 O_NONBLOCK、O_CLOEXEC）需要通过 fcntl() 额外设置。

2.   pipe2()

-   原型：int pipe2(int pipefd[2], int flags);
-   是 pipe() 的增强版，允许在创建管道时直接指定标志（如 O_NONBLOCK、O_CLOEXEC、O_DIRECT 等）。
-   更高效、更安全，避免了竞态条件（race condition），因为标志在创建时就已设置。

## 文件系统注册

```c
// 超级块操作结构
static const struct super_operations pipefs_ops = {
  // 销毁inode时调用的函数，这里使用free_inode_nonrcu函数来释放inode结构体
	.destroy_inode = free_inode_nonrcu, 
  // 获取文件系统统计信息时调用
	.statfs = simple_statfs,
};

// 初始化 pipefs 的文件系统上下文
static int pipefs_init_fs_context(struct fs_context *fc)
{
  // 创建一个伪文件系统的上下文
	struct pseudo_fs_context *ctx = init_pseudo(fc, PIPEFS_MAGIC);
	if (!ctx)
		return -ENOMEM;
	ctx->ops = &pipefs_ops; // 设置超级块操作
	ctx->dops = &pipefs_dentry_operations; // dentry的操作
	return 0;
}

// 文件系统的类型
static struct file_system_type pipe_fs_type = {
	.name		= "pipefs",
	.init_fs_context = pipefs_init_fs_context,
	.kill_sb	= kill_anon_super,
};

static int __init init_pipe_fs(void)
{
	int err = register_filesystem(&pipe_fs_type);

	if (!err) {
    // 获取一个vfsmount结构的数据, 记录在全局变量里面
		pipe_mnt = kern_mount(&pipe_fs_type);
		if (IS_ERR(pipe_mnt)) {
			err = PTR_ERR(pipe_mnt);
			unregister_filesystem(&pipe_fs_type);
		}
	}
	return err;
}
// 系统初始化的时候调用
fs_initcall(init_pipe_fs);
```

这个文件系统用于不会被mount, 否则会带来安全隐患, 因为不需要用户交互，所以不需要在文件系统根目录实现操作
