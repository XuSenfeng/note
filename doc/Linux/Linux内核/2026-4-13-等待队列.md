# 等待队列

## 数据类型

+   等待队列管理

```c
struct wait_queue_head {
	spinlock_t		lock;
	struct list_head	head;
};
```

+   一个等待的项

```c
/*
 * A single wait-queue entry structure:
 */
struct wait_queue_entry {
	unsigned int		flags; //  该等待项的标志位
	void			*private; // 这个等待项关联的对象。通常是当前任务的指针
	/*
	唤醒回调函数。
  当有 wakeup 发生时，内核会调用这个函数决定：
  是否唤醒当前等待者
  唤醒后要不要移出队列
  返回值是否表示成功唤醒
	*/
  wait_queue_func_t	func;
	struct list_head	entry;
};
```

-   WQ_FLAG_EXCLUSIVE：独占等待者
-   WQ_FLAG_WOKEN：已被唤醒
-   WQ_FLAG_BOOKMARK：遍历书签, 如果一次遍历过长（超过 `WAITQUEUE_WALK_BREAK_CNT`），就把一个临时节点插到当前位置后面，并打上 `WQ_FLAG_BOOKMARK`。
-   WQ_FLAG_CUSTOM：自定义唤醒函数相关
-   WQ_FLAG_DONE：已完成

### 数据初始化

```c
void init_wait_entry(struct wait_queue_entry *wq_entry, int flags)
{
	wq_entry->flags = flags;
	wq_entry->private = current;
	wq_entry->func = autoremove_wake_function;
	INIT_LIST_HEAD(&wq_entry->entry);
}
// 默认的唤醒处理函数
__sched int autoremove_wake_function(struct wait_queue_entry *wq_entry, unsigned int mode,
				     int sync, void *key)
{
  // 尝试进行唤醒
	int ret = default_wake_function(wq_entry, mode, sync, key);

	if (ret)
    // 成功的话把他的等待项删除
		list_del_init_careful(&wq_entry->entry);

	return ret;
}
```



## 等待事件

```c
/**
 * wait_event - sleep until a condition gets true
 * @wq_head: the waitqueue to wait on
 * @condition: a C expression for the event to wait for
 *
 * The process is put to sleep (TASK_UNINTERRUPTIBLE) until the
 * @condition evaluates to true. The @condition is checked each time
 * the waitqueue @wq_head is woken up.
 *
 * wake_up() has to be called after changing any variable that could
 * change the result of the wait condition.
 */

#define wait_event(wq_head, condition)						\
do {										\
	might_sleep();								\
	if (condition)								\
		break;								\
	__wait_event(wq_head, condition);					\
} while (0)

#define __wait_event(wq_head, condition)					\
	(void)___wait_event(wq_head, condition, TASK_UNINTERRUPTIBLE, 0, 0,	\
			    schedule())

#define ___wait_event(wq_head, condition, state, exclusive, ret, cmd)		\
({										\
	__label__ __out;							\
	/* 创建一个实际挂载使用的变量 */  							\
	struct wait_queue_entry __wq_entry;					\
	long __ret = ret;	/* explicit shadow */				\
										\
	init_wait_entry(&__wq_entry, exclusive ? WQ_FLAG_EXCLUSIVE : 0);	\
	for (;;) {								\
    /*把当前进程加入等待队列, 当前进程的睡眠状态, 有信号的时候退出返回错误 */   \
		long __int = prepare_to_wait_event(&wq_head, &__wq_entry, state);\
										\
		if (condition)							\
			break;							\
										\
		if (___wait_is_interruptible(state) && __int) {			\
			__ret = __int;						\
			goto __out;						\
		}								\
		/* 进行调度 */								\
		cmd;								\
	}									\
	finish_wait(&wq_head, &__wq_entry);					\
__out:	__ret;									\
})
```

默认的等待是一个不可以呗中断的睡眠, 不响应信号

### 可以被打断的

所以实际使用的时候可以自己实现这个过程

```c
static ssize_t xxx_write(struct file *file, const char *buffer, size_t count, loff_t *ppos)
{
    ssize_t ret = 0;
    int avail;
    DECLARE_WAITQUEUE(wait, current);  /* 定义等待队列项，关联当前进程 */

    add_wait_queue(&xxx_wait, &wait);   /* 将当前进程加入等待队列头 */

    ret = count;  /* 预设返回值为写入字节数 */

    /* 循环等待设备缓冲区可写 */
    do {
        avail = device_writable(...);  /* 检查设备可写空间（需补全具体参数） */
        
        if (avail < 0) {
            set_current_state(TASK_INTERRUPTIBLE);  /* 设置进程为可中断睡眠状态 */
        }

        if (avail < 0) {
            if (file->f_flags & O_NONBLOCK) {  /* 非阻塞模式：直接返回 */
                if (!ret)
                    ret = -EAGAIN;
                goto out;
            }
            schedule();  /* 主动调度，让出CPU */
            
            if (signal_pending(current)) {  /* 因信号唤醒：处理信号 */
                if (!ret)
                    ret = -ERESTARTSYS;
                goto out;
            }
        }
    } while (avail < 0);

    /* 设备可写：执行实际写操作（需补全具体参数） */
    device_write(...);

out:
    remove_wait_queue(&xxx_wait, &wait);  /* 从等待队列头移除当前进程 */
    set_current_state(TASK_RUNNING);       /* 强制设置进程为运行态 */
    return ret;
}

int default_wake_function(wait_queue_entry_t *curr, unsigned mode, int wake_flags,
			  void *key)
{
	WARN_ON_ONCE(IS_ENABLED(CONFIG_SCHED_DEBUG) && wake_flags & ~(WF_SYNC | WF_ANDROID_VENDOR));
	return try_to_wake_up(curr->private, mode, wake_flags);
}
```

## 唤醒

```c
#define wake_up(x)			__wake_up(x, TASK_NORMAL, 1, NULL)
void __wake_up(struct wait_queue_head *wq_head, unsigned int mode,
			int nr_exclusive, void *key)
{
	__wake_up_common_lock(wq_head, mode, nr_exclusive, 0, key);
}

static void __wake_up_common_lock(struct wait_queue_head *wq_head, unsigned int mode,
			int nr_exclusive, int wake_flags, void *key)
{
	unsigned long flags;
	// 创建一个临时的 wait_queue_entry_t 结构体变量 bookmark，并初始化其成员。
	wait_queue_entry_t bookmark;

	bookmark.flags = 0;
	bookmark.private = NULL;
	bookmark.func = NULL;
	INIT_LIST_HEAD(&bookmark.entry);

	do {
		spin_lock_irqsave(&wq_head->lock, flags);
		// 实际的唤醒操作由 __wake_up_common() 函数完成，传入的参数包括等待队列头、
		// 唤醒模式、独占唤醒数量、唤醒标志、键值和书签。
		nr_exclusive = __wake_up_common(wq_head, mode, nr_exclusive,
						wake_flags, key, &bookmark);
		spin_unlock_irqrestore(&wq_head->lock, flags);
	} while (bookmark.flags & WQ_FLAG_BOOKMARK);
}

static int __wake_up_common(struct wait_queue_head *wq_head, unsigned int mode,
			int nr_exclusive, int wake_flags, void *key,
			wait_queue_entry_t *bookmark)
{
	wait_queue_entry_t *curr, *next;
	int cnt = 0;

	lockdep_assert_held(&wq_head->lock);
	// 获取下一个等待队列项，如果 bookmark 不为 NULL 且 bookmark 的 flags 包含 
	// WQ_FLAG_BOOKMARK 标志，则将 curr 设置为 bookmark 的下一个等待队列项，
	// 并将 bookmark 从等待队列中删除并清除其 flags。
	if (bookmark && (bookmark->flags & WQ_FLAG_BOOKMARK)) {
		curr = list_next_entry(bookmark, entry);

		list_del(&bookmark->entry);
		bookmark->flags = 0;
	} else
		curr = list_first_entry(&wq_head->head, wait_queue_entry_t, entry);

	// 如果 curr 是等待队列头部的第一个项，则直接返回 nr_exclusive。
	if (&curr->entry == &wq_head->head)
		return nr_exclusive;

	// 遍历等待队列，从 curr 开始，直到遇到等待队列头部为止。对于每个等待队列项，获取其 flags，
	// 并调用其 func 函数进行唤醒操作。
	list_for_each_entry_safe_from(curr, next, &wq_head->head, entry) {
		unsigned flags = curr->flags;
		int ret;

		if (flags & WQ_FLAG_BOOKMARK)
			continue;

		ret = curr->func(curr, mode, wake_flags, key);
		if (ret < 0)
			break;
		if (ret && (flags & WQ_FLAG_EXCLUSIVE) && !--nr_exclusive)
			// 唤醒成功且当前项是独占项，并且已经唤醒了足够数量的独占项时，
			break;

		if (bookmark && (++cnt > WAITQUEUE_WALK_BREAK_CNT) &&
				(&next->entry != &wq_head->head)) {
			// 如果 bookmark 不为 NULL 且已经遍历了足够数量的项，
			// 并且下一个项不是等待队列头部，则将 bookmark 插入到当前项和下一个项之间，
			bookmark->flags = WQ_FLAG_BOOKMARK;
			list_add_tail(&bookmark->entry, &next->entry);
			break;
		}
	}

	return nr_exclusive;
}
```
