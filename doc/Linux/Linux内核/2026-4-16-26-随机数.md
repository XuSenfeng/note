# 随机数

drivers/char/random.c

随机数子系统不是“凭空生随机数”，而是把很多不容易预测的系统噪声收集起来，混合成一个密码学随机源

它大致分成三层：

-   熵收集：把时间抖动、中断时序、启动过程信息、硬件随机源等输入到池里。
-   熵池混合：把这些输入不断哈希/混合，避免原始值直接暴露。
-   CRNG 输出：当种子足够了，再用 ChaCha 这类流密码从内部状态扩展出大量随机字节。



## 实际实现

这个是读取的入口函数

```c
static ssize_t random_read_iter(struct kiocb *kiocb, struct iov_iter *iter)
{
	int ret;
	
  // 检查一下随机数生成器是不是已经初始化了
	if (!crng_ready() &&
	    ((kiocb->ki_flags & (IOCB_NOWAIT | IOCB_NOIO)) ||
	     (kiocb->ki_filp->f_flags & O_NONBLOCK)))
		return -EAGAIN;
	
  // 等待 crng_init_wait 这个等待队列
	ret = wait_for_random_bytes();
	if (ret != 0)
		return ret;
	return get_random_bytes_user(iter);
}

int wait_for_random_bytes(void)
{
	while (!crng_ready()) {
		// 随机数生成器还没有准备好，等待随机数生成器准备好
		int ret;

		// 尝试生成随机数，以便尽快让随机数生成器准备好
		try_to_generate_entropy();
		ret = wait_event_interruptible_timeout(crng_init_wait, crng_ready(), HZ);
		if (ret)
			return ret > 0 ? 0 : ret;
	}
	return 0;
}

static void __cold try_to_generate_entropy(void)
{
	struct {
		unsigned long entropy;
		struct timer_list timer;
	} stack;
	// 获取一个很快的时间/周期计数器读数
	stack.entropy = random_get_entropy();

	/* Slow counter - or none. Don't even bother */
	if (stack.entropy == random_get_entropy())
		// 再次获取, 相同的数字说面没有熵, 直接返回
		return;

	// 设置定时器
	timer_setup_on_stack(&stack.timer, entropy_timer, 0);
	while (!crng_ready() && !signal_pending(current)) {
		if (!timer_pending(&stack.timer))
			mod_timer(&stack.timer, jiffies + 1);
		mix_pool_bytes(&stack.entropy, sizeof(stack.entropy));
		schedule();
		stack.entropy = random_get_entropy();
	}

	del_timer_sync(&stack.timer);
	destroy_timer_on_stack(&stack.timer);
	mix_pool_bytes(&stack.entropy, sizeof(stack.entropy));
}

static void __cold entropy_timer(struct timer_list *t)
{
	credit_init_bits(1); // 每次定时器触发, 认为我们至少获得了1 bit的熵
}
// 如果 crng_ready() 为假，就调用 _credit_init_bits(bits)
// 系统启动早期，内核还在积累熵，这时需要把一些输入来源“记到熵池里”，推动 crng_init 变成 ready
// 并在累计到阈值后触发后续初始化动作
#define credit_init_bits(bits) if (!crng_ready()) _credit_init_bits(bits)
```
