## spidev：SPI“万能”驱动

+   需要注意两个结构体, `spi_device`以及`spi_driver`
+   `spi_device`是读取设备树和驱动匹配的时候自动生成的
+   `spi_driver`是实际需要自己实现的部分, 里面的probe是在匹配以后执行的函数, 替代`platform_driver`, 在probe函数参数可以获得实际使用的`spi_device`
+   初始化`spi_message`结构体收发消息=>`spi_message_init`, 里面的每一个信息使用`spi_transfer`描述
+   添加具体的信息(`spi_transfer`)到`spi_message`里面`spi_message_add_tail`
+   把信息添加到队列里面=>`spi_message_add_tail`
+   同步异步传输=>`spi_sync/spi_async`

>   上面的传输可以使用`spi_write`以及`spi_read`函数替代
>
>   在使用官方的cs软件片选的时候, 一次操作不要分为多个函数, 否则片选引脚拉低会中断, , 可以使用`spi_write_then_read`函数

```c
static int spidev_probe(struct spi_device *spi)
{
	struct spidev_data	*spidev;
	int			status;
	unsigned long		minor;
	...
	/* Allocate driver data */
	spidev = kzalloc(sizeof(*spidev), GFP_KERNEL);
	if (!spidev)
		return -ENOMEM;

	/* Initialize the driver data */
    // 初始化基础的参数
	spidev->spi = spi;
	spin_lock_init(&spidev->spi_lock);
	mutex_init(&spidev->buf_lock);

	INIT_LIST_HEAD(&spidev->device_entry);

	/* If we can allocate a minor number, hook up this device.
	 * Reusing minors is fine so long as udev or mdev is working.
	 */
	mutex_lock(&device_list_lock);
    // 分配一个次设备号
	minor = find_first_zero_bit(minors, N_SPI_MINORS);
	if (minor < N_SPI_MINORS) {
		struct device *dev;
		// 记录实际的设备号
		spidev->devt = MKDEV(SPIDEV_MAJOR, minor);
        // 在class创建一个新的设备
		dev = device_create(spidev_class, &spi->dev, spidev->devt,
				    spidev, "spidev%d.%d",
				    spi->master->bus_num, spi->chip_select); // 使用设备的总线以及片选进行命名
		status = PTR_ERR_OR_ZERO(dev);
	} else {
		dev_dbg(&spi->dev, "no minor number available!\n");
		status = -ENODEV;
	}
	if (status == 0) {
		set_bit(minor, minors);
		list_add(&spidev->device_entry, &device_list);
	}
	mutex_unlock(&device_list_lock);

	spidev->speed_hz = spi->max_speed_hz; // 初始化传输速率

	if (status == 0)
        // 配置设置
		spi_set_drvdata(spi, spidev);
	else
		kfree(spidev);

	return status;
}
```

```c
static int spidev_open(struct inode *inode, struct file *filp)
{
	struct spidev_data	*spidev;
	int			status = -ENXIO;

	mutex_lock(&device_list_lock);
	// 记录所有spi设备的链表
	list_for_each_entry(spidev, &device_list, device_entry) {
		if (spidev->devt == inode->i_rdev) {
			status = 0;
			break;
		}
	}

	if (status) {
		pr_debug("spidev: nothing for minor %d\n", iminor(inode));
		goto err_find_dev;
	}
	// buf的分配
	if (!spidev->tx_buffer) {
		spidev->tx_buffer = kmalloc(bufsiz, GFP_KERNEL);
		if (!spidev->tx_buffer) {
			dev_dbg(&spidev->spi->dev, "open/ENOMEM\n");
			status = -ENOMEM;
			goto err_find_dev;
		}
	}

	if (!spidev->rx_buffer) {
		spidev->rx_buffer = kmalloc(bufsiz, GFP_KERNEL);
		if (!spidev->rx_buffer) {
			dev_dbg(&spidev->spi->dev, "open/ENOMEM\n");
			status = -ENOMEM;
			goto err_alloc_rx_buf;
		}
	}
	// 记录在设备文件系统里面
	spidev->users++;
	filp->private_data = spidev;
	nonseekable_open(inode, filp); // 不支持lseek指针

	mutex_unlock(&device_list_lock);
	return 0;

err_alloc_rx_buf:
	kfree(spidev->tx_buffer);
	spidev->tx_buffer = NULL;
err_find_dev:
	mutex_unlock(&device_list_lock);
	return status;
}
```

#### spidev_read()函数

```c
static ssize_t
spidev_read(struct file *filp, char __user *buf, size_t count, loff_t *f_pos)
{
	struct spidev_data	*spidev;
	ssize_t			status = 0;

	/* chipselect only toggles at start or end of operation */
    // 读取的数据是不可以大于部分的
	if (count > bufsiz)
		return -EMSGSIZE;

	spidev = filp->private_data;

	mutex_lock(&spidev->buf_lock);
	status = spidev_sync_read(spidev, count); // 同步读取
	if (status > 0) {
		unsigned long	missing;

		missing = copy_to_user(buf, spidev->rx_buffer, status);
		if (missing == status)
			status = -EFAULT;
		else
			status = status - missing;
	}
	mutex_unlock(&spidev->buf_lock);

	return status;
}
```

#### spidev_ioctl()函数

```c
static long
spidev_ioctl(struct file *filp, unsigned int cmd, unsigned long arg)
{
	int			retval = 0;
	struct spidev_data	*spidev;
	struct spi_device	*spi;
	u32			tmp;
	unsigned		n_ioc;
	struct spi_ioc_transfer	*ioc;
	...
	spidev = filp->private_data;
	spin_lock_irq(&spidev->spi_lock);
	spi = spi_dev_get(spidev->spi);
	spin_unlock_irq(&spidev->spi_lock);
	...
     // 处理不同的控制命令
	switch (cmd) {
	/* read requests */
	case SPI_IOC_RD_MODE: // 获取模式
		retval = put_user(spi->mode & SPI_MODE_MASK,
					(__u8 __user *)arg);
		break;
	case SPI_IOC_RD_MODE32:
		retval = put_user(spi->mode & SPI_MODE_MASK,
					(__u32 __user *)arg);
		break;
	case SPI_IOC_RD_LSB_FIRST:
		retval = put_user((spi->mode & SPI_LSB_FIRST) ?  1 : 0,
					(__u8 __user *)arg);
		break;
	case SPI_IOC_RD_BITS_PER_WORD:
		retval = put_user(spi->bits_per_word, (__u8 __user *)arg);
		break;
	case SPI_IOC_RD_MAX_SPEED_HZ:
		retval = put_user(spidev->speed_hz, (__u32 __user *)arg);
		break;

	/* write requests */
	case SPI_IOC_WR_MODE:
	case SPI_IOC_WR_MODE32:
		if (cmd == SPI_IOC_WR_MODE)
			retval = get_user(tmp, (u8 __user *)arg);
		else
			retval = get_user(tmp, (u32 __user *)arg);
		if (retval == 0) {
			u32	save = spi->mode;

			if (tmp & ~SPI_MODE_MASK) {
				retval = -EINVAL;
				break;
			}

			tmp |= spi->mode & ~SPI_MODE_MASK;
			spi->mode = (u16)tmp;
			retval = spi_setup(spi);
			if (retval < 0)
				spi->mode = save;
			else
				dev_dbg(&spi->dev, "spi mode %x\n", tmp);
		}
		break;
	case SPI_IOC_WR_LSB_FIRST:
		retval = get_user(tmp, (__u8 __user *)arg);
		if (retval == 0) {
			u32	save = spi->mode;

			if (tmp)
				spi->mode |= SPI_LSB_FIRST;
			else
				spi->mode &= ~SPI_LSB_FIRST;
			retval = spi_setup(spi);
			if (retval < 0)
				spi->mode = save;
			else
				dev_dbg(&spi->dev, "%csb first\n",
						tmp ? 'l' : 'm');
		}
		break;
	case SPI_IOC_WR_BITS_PER_WORD:
		retval = get_user(tmp, (__u8 __user *)arg);
		if (retval == 0) {
			u8	save = spi->bits_per_word;

			spi->bits_per_word = tmp;
			retval = spi_setup(spi);
			if (retval < 0)
				spi->bits_per_word = save;
			else
				dev_dbg(&spi->dev, "%d bits per word\n", tmp);
		}
		break;
	case SPI_IOC_WR_MAX_SPEED_HZ:
		retval = get_user(tmp, (__u32 __user *)arg);
		if (retval == 0) {
			u32	save = spi->max_speed_hz;

			spi->max_speed_hz = tmp;
			retval = spi_setup(spi);
			if (retval >= 0)
				spidev->speed_hz = tmp;
			else
				dev_dbg(&spi->dev, "%d Hz (max)\n", tmp);
			spi->max_speed_hz = save;
		}
		break;

	default:
            // 消息的全双工通信, 可以使用短接进行消息的回环
		/* segmented and/or full-duplex I/O request */
		/* Check message and copy into scratch area */
		ioc = spidev_get_ioc_message(cmd,
				(struct spi_ioc_transfer __user *)arg, &n_ioc);
		if (IS_ERR(ioc)) {
			retval = PTR_ERR(ioc);
			break;
		}
		if (!ioc)
			break;	/* n_ioc is also 0 */

		/* translate to spi_message, execute */
		retval = spidev_message(spidev, ioc, n_ioc);
		kfree(ioc);
		break;
	}

	mutex_unlock(&spidev->buf_lock);
	spi_dev_put(spi);
	return retval;
}
```

```c
static struct spi_ioc_transfer *
spidev_get_ioc_message(unsigned int cmd, struct spi_ioc_transfer __user *u_ioc,
		unsigned *n_ioc)
{
	u32	tmp;

	/* Check type, command number and direction */
    // 检查命令是否非法
	if (_IOC_TYPE(cmd) != SPI_IOC_MAGIC
			|| _IOC_NR(cmd) != _IOC_NR(SPI_IOC_MESSAGE(0))
			|| _IOC_DIR(cmd) != _IOC_WRITE)
		return ERR_PTR(-ENOTTY);

	tmp = _IOC_SIZE(cmd); // cmd里面记录了spi_ioc_transfer数组的大小
	if ((tmp % sizeof(struct spi_ioc_transfer)) != 0)
		return ERR_PTR(-EINVAL);
	*n_ioc = tmp / sizeof(struct spi_ioc_transfer);
	if (*n_ioc == 0)
		return NULL;

	/* copy into scratch area */
	return memdup_user(u_ioc, tmp);
}
```

##### spidev_message()函数

内核处理发送信息的函数, 用于处理用户空间通过 `ioctl` 接口发起的 SPI 数据传输请求

该函数实现了用户空间与内核 SPI 子系统之间的 “桥梁” 作用

```c
static int spidev_message(struct spidev_data *spidev,
		struct spi_ioc_transfer *u_xfers, unsigned n_xfers)
{
	struct spi_message	msg;
	struct spi_transfer	*k_xfers;
	struct spi_transfer	*k_tmp;
	struct spi_ioc_transfer *u_tmp;
	unsigned		n, total, tx_total, rx_total;
	u8			*tx_buf, *rx_buf;
	int			status = -EFAULT;
	// 初始化message
	spi_message_init(&msg);
	k_xfers = kcalloc(n_xfers, sizeof(*k_tmp), GFP_KERNEL);
	if (k_xfers == NULL)
		return -ENOMEM;

	tx_buf = spidev->tx_buffer;
	rx_buf = spidev->rx_buffer;
	total = 0;
	tx_total = 0;
	rx_total = 0;
	for (n = n_xfers, k_tmp = k_xfers, u_tmp = u_xfers;
			n;
			n--, k_tmp++, u_tmp++) {
		k_tmp->len = u_tmp->len;

		total += k_tmp->len;
		
		if (total > INT_MAX || k_tmp->len > INT_MAX) {
			status = -EMSGSIZE;
			goto done;
		}

		if (u_tmp->rx_buf) {
			/* this transfer needs space in RX bounce buffer */
			rx_total += k_tmp->len;
			if (rx_total > bufsiz) {
				status = -EMSGSIZE;
				goto done;
			}
			k_tmp->rx_buf = rx_buf;
			rx_buf += k_tmp->len;
		}
		if (u_tmp->tx_buf) {
			/* this transfer needs space in TX bounce buffer */
			tx_total += k_tmp->len;
			if (tx_total > bufsiz) {
				status = -EMSGSIZE;
				goto done;
			}
			k_tmp->tx_buf = tx_buf;
            // 获取一下需要发送的消息
			if (copy_from_user(tx_buf, (const u8 __user *)
						(uintptr_t) u_tmp->tx_buf,
					u_tmp->len))
				goto done;
			tx_buf += k_tmp->len;
		}

		k_tmp->cs_change = !!u_tmp->cs_change;
		k_tmp->tx_nbits = u_tmp->tx_nbits;
		k_tmp->rx_nbits = u_tmp->rx_nbits;
		k_tmp->bits_per_word = u_tmp->bits_per_word;
		k_tmp->delay_usecs = u_tmp->delay_usecs;
		k_tmp->speed_hz = u_tmp->speed_hz;
		if (!k_tmp->speed_hz)
			k_tmp->speed_hz = spidev->speed_hz;
		spi_message_add_tail(k_tmp, &msg);
	}
	// 同步发送时
	status = spidev_sync(spidev, &msg);
	if (status < 0)
		goto done;

	/* copy any rx data out of bounce buffer */
	rx_buf = spidev->rx_buffer;
	for (n = n_xfers, u_tmp = u_xfers; n; n--, u_tmp++) {
		if (u_tmp->rx_buf) {
            // 返回读取的消息
			if (copy_to_user((u8 __user *)
					(uintptr_t) u_tmp->rx_buf, rx_buf,
					u_tmp->len)) {
				status = -EFAULT;
				goto done;
			}
			rx_buf += u_tmp->len;
		}
	}
	status = total;

done:
	kfree(k_xfers);
	return status;
}
```

### 用户使用代码

```c
#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <linux/spi/spidev.h>

#define SPI_DEV "/dev/spidev0.0"  // SPI 设备节点

int main() {
    int fd;
    int ret;
    uint8_t tx_buf[2] = {0x01, 0x02};  // 发送数据
    uint8_t rx_buf[2] = {0};            // 接收数据

    // 打开 SPI 设备
    fd = open(SPI_DEV, O_RDWR);
    if (fd < 0) {
        perror("Failed to open SPI device");
        return 1;
    }

    // 配置 SPI 模式（CPOL/CPHA）
    uint8_t mode = SPI_MODE_0;
    ret = ioctl(fd, SPI_IOC_WR_MODE, &mode);
    if (ret < 0) {
        perror("Failed to set SPI mode");
        close(fd);
        return 1;
    }

    // 配置 SPI 时钟频率
    uint32_t speed = 1000000;  // 1MHz
    ret = ioctl(fd, SPI_IOC_WR_MAX_SPEED_HZ, &speed);
    if (ret < 0) {
        perror("Failed to set SPI speed");
        close(fd);
        return 1;
    }

    // 配置 SPI 位宽
    uint8_t bits = 8;
    ret = ioctl(fd, SPI_IOC_WR_BITS_PER_WORD, &bits);
    if (ret < 0) {
        perror("Failed to set SPI bits per word");
        close(fd);
        return 1;
    }

    // 构造 SPI 传输请求
    struct spi_ioc_transfer xfer = {
        .tx_buf = (unsigned long)tx_buf,
        .rx_buf = (unsigned long)rx_buf,
        .len = 2,  // 传输 2 字节
        .delay_usecs = 0,
        .speed_hz = speed,
        .bits_per_word = bits,
    };

    // 执行 SPI 传输
    ret = ioctl(fd, SPI_IOC_MESSAGE(1), &xfer);
    if (ret < 0) {
        perror("Failed to perform SPI transfer");
        close(fd);
        return 1;
    }

    // 打印接收到的数据
    printf("Received data: 0x%02x 0x%02x\n", rx_buf[0], rx_buf[1]);

    // 关闭 SPI 设备
    close(fd);
    return 0;
}
```

## SPI 字符设备

```c
#include <linux/module.h>
#include <linux/spi/spi.h>
#include <linux/cdev.h>
#include <linux/fs.h>
#include <linux/uaccess.h>
#include <linux/delay.h>

// 设备名称和数量
#define DEVICE_NAME "sht3x_temp"
#define DEVICE_COUNT 1

// 传感器私有数据结构
struct sht3x_data {
    struct spi_device *spi;  // SPI设备指针
    struct cdev cdev;         // 字符设备结构体
    dev_t devno;              // 设备号
    struct class *class;      // 设备类（用于自动创建设备节点）
};

// 读取温湿度的函数（SPI通信核心）
static int sht3x_read_temp_humidity(struct sht3x_data *data, u16 *temp, u16 *humidity)
{
    int ret;
    u8 tx_buf[2] = {0x2C, 0x06};  // SHT3x测量命令（高精度）
    u8 rx_buf[6] = {0};            // 接收数据：温度(2B)+CRC(1B)+湿度(2B)+CRC(1B)

    // SPI通信：先写命令，再读数据
    ret = spi_write_then_read(data->spi, tx_buf, 2, rx_buf, 6);
    if (ret < 0) {
        dev_err(&data->spi->dev, "SPI read failed: %d\n", ret);
        return ret;
    }

    // 解析温度和湿度（SHT3x数据格式）
    *temp = (rx_buf[0] << 8) | rx_buf[1];
    *humidity = (rx_buf[3] << 8) | rx_buf[4];

    return 0;
}

// 字符设备的read函数（用户空间读数据时调用）
static ssize_t sht3x_read(struct file *file, char __user *buf, size_t count, loff_t *ppos)
{
    struct sht3x_data *data = file->private_data;
    u16 temp_raw, humidity_raw;
    char output[64];
    int len;
    int ret;

    // 只允许读一次（避免重复读取）
    if (*ppos > 0)
        return 0;

    // 读取传感器数据
    ret = sht3x_read_temp_humidity(data, &temp_raw, &humidity_raw);
    if (ret < 0)
        return ret;

    // 转换为实际温度（℃）和湿度（%RH）
    float temp = -45.0 + 175.0 * ((float)temp_raw / 65535.0);
    float humidity = 100.0 * ((float)humidity_raw / 65535.0);

    // 格式化输出字符串
    len = snprintf(output, sizeof(output), "Temp: %.2f C, Humidity: %.2f %%RH\n", temp, humidity);

    // 复制数据到用户空间
    if (copy_to_user(buf, output, len))
        return -EFAULT;

    *ppos = len;
    return len;
}

// 字符设备的open函数
static int sht3x_open(struct inode *inode, struct file *file)
{
    // 从inode中获取私有数据
    struct sht3x_data *data = container_of(inode->i_cdev, struct sht3x_data, cdev);
    file->private_data = data;  // 保存到file中，供read使用
    return 0;
}

// 字符设备操作函数集
static const struct file_operations sht3x_fops = {
    .owner = THIS_MODULE,
    .open = sht3x_open,
    .read = sht3x_read,
};

// SPI驱动的probe函数（设备和驱动匹配时调用）
static int sht3x_probe(struct spi_device *spi)
{
    struct sht3x_data *data;
    int ret;

    // 1. 分配私有数据
    data = devm_kzalloc(&spi->dev, sizeof(*data), GFP_KERNEL);
    if (!data)
        return -ENOMEM;

    // 2. 保存SPI设备指针
    data->spi = spi;
    spi_set_drvdata(spi, data);  // 将私有数据关联到SPI设备

    // 3. 分配设备号
    ret = alloc_chrdev_region(&data->devno, 0, DEVICE_COUNT, DEVICE_NAME);
    if (ret < 0) {
        dev_err(&spi->dev, "Failed to allocate devno: %d\n", ret);
        return ret;
    }

    // 4. 初始化字符设备
    cdev_init(&data->cdev, &sht3x_fops);
    data->cdev.owner = THIS_MODULE;

    // 5. 添加字符设备到内核
    ret = cdev_add(&data->cdev, data->devno, DEVICE_COUNT);
    if (ret < 0) {
        dev_err(&spi->dev, "Failed to add cdev: %d\n", ret);
        goto err_unregister_devno;
    }

    // 6. 创建设备类（用于自动创建设备节点）
    data->class = class_create(THIS_MODULE, DEVICE_NAME);
    if (IS_ERR(data->class)) {
        ret = PTR_ERR(data->class);
        dev_err(&spi->dev, "Failed to create class: %d\n", ret);
        goto err_del_cdev;
    }

    // 7. 创建设备节点（/dev/sht3x_temp）
    device_create(data->class, NULL, data->devno, NULL, DEVICE_NAME);
    if (IS_ERR(data->class)) {
        ret = PTR_ERR(data->class);
        dev_err(&spi->dev, "Failed to create device: %d\n", ret);
        goto err_destroy_class;
    }

    dev_info(&spi->dev, "SHT3x driver probed successfully\n");
    return 0;

// 错误处理（回滚操作）
err_destroy_class:
    class_destroy(data->class);
err_del_cdev:
    cdev_del(&data->cdev);
err_unregister_devno:
    unregister_chrdev_region(data->devno, DEVICE_COUNT);
    return ret;
}

// SPI驱动的remove函数（设备移除时调用）
static void sht3x_remove(struct spi_device *spi)
{
    struct sht3x_data *data = spi_get_drvdata(spi);

    // 销毁设备节点
    device_destroy(data->class, data->devno);
    // 销毁设备类
    class_destroy(data->class);
    // 删除字符设备
    cdev_del(&data->cdev);
    // 释放设备号
    unregister_chrdev_region(data->devno, DEVICE_COUNT);

    dev_info(&spi->dev, "SHT3x driver removed\n");
}

// 设备树匹配表（用于和设备树节点匹配）
static const struct of_device_id sht3x_of_match[] = {
    { .compatible = "sensirion,sht3x" },
    { }
};
MODULE_DEVICE_TABLE(of, sht3x_of_match);

// SPI驱动结构体
static struct spi_driver sht3x_driver = {
    .probe = sht3x_probe,
    .remove = sht3x_remove,
    .driver = {
        .name = "sht3x_spi",
        .of_match_table = sht3x_of_match,
    },
};
module_spi_driver(sht3x_driver);  // 注册SPI驱动

MODULE_LICENSE("GPL");
MODULE_AUTHOR("Your Name");
MODULE_DESCRIPTION("SHT3x SPI Temperature/Humidity Character Device Driver");
```
