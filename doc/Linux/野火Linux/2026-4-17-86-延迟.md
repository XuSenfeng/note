# 延迟

## 短延迟

```c
void ndelay(unsigned long nsecs);
void udelay(unsigned long usecs);
void mdelay(unsigned long msecs);

static inline void ndelay(unsigned long x)
{
  // 实际是一个向上取整的微妙, 架构里面实现这个的话, 这里不会再使用
	udelay(DIV_ROUND_UP(x, 1000));
}
#define ndelay(x) ndelay(x)

#define mdelay(n) (\
	(__builtin_constant_p(n) && (n)<=MAX_UDELAY_MS) ? udelay((n)*1000) : \
	({unsigned long __ms=(n); while (__ms--) udelay(1000);}))

// 实际的udelay是在不同的架构里面实现
// arm64  include/asm-generic/delay.h
#define udelay(n)							\
	({								\
		if (__builtin_constant_p(n)) {				\// 是否是常量
			if ((n) / 20000 >= 1)				\// 如果常量太大
				 __bad_udelay();			\// 故意触发编译期错误，防止不合理的大 busy-wait
			else						\// 对常量参数的快速路径, 把“微秒”换成 __const_udelay() 需要的 xloops 标度
				__const_udelay((n) * 0x10c7ul);		\
		} else {						\
			__udelay(n);					\
		}							\
	})

void __delay(unsigned long cycles)
{
	cycles_t start = get_cycles();

	if (arch_timer_evtstrm_available()) {
    // 第一段循环（可选，省电等待）, 等待时尽量低功耗，不是纯空转
		const cycles_t timer_evt_period =
			USECS_TO_CYCLES(ARCH_TIMER_EVT_STREAM_PERIOD_US);

		while ((get_cycles() - start + timer_evt_period) < cycles)
			wfe();
	}

	while ((get_cycles() - start) < cycles)
    // 用 cpu_relax 持续轮询直到时间到
		// 执行一个架构相关的指令, 以减少CPU的功耗, 同时也可以让CPU更快地响应中断. 
		// 在这个循环中在等待的同时, 处理其他任务或者响应中断, 从而提高系统的效率.
		cpu_relax(); 
}

inline void __const_udelay(unsigned long xloops)
{
	__delay(xloops_to_cycles(xloops));
}

static inline unsigned long xloops_to_cycles(unsigned long xloops)
{
	return (xloops * loops_per_jiffy * HZ) >> 32;
}
```

>   `__builtin_constant_p(x)` 是 GCC/Clang 提供的一个编译器内建判断：
>
>   -   作用：判断表达式 `x` 在当前编译阶段是否能被当作编译期常量。
>   -   返回：能确定为常量时返回 `1`，否则返回 `0`。
>   -   特点：这是“编译器优化层面的判断”，结果会受上下文和优化影响，不等同于 C 语言语义里的“必须是常量表达式”。
>
>   -   常量小延迟时让编译器直接优化成更高效代码。
>   -   非常量或较大值时避免一次性 `udelay` 可能带来的溢出风

xloop: “延时内部的标准化计数单位”，不是直接的“循环次数”, `xloops = n * 0x10c7ul`, 再根据机器速度换成真实等待量, 而是用于跨架构统一计算延时的“中间标尺”

内核在启动时，会运行一个延迟测试程序（delay loop calibration），计算出lpj （loops per jiffy）, 让后面的`udelay/ndelay`忙等延时“尽量接近真实时间”。

```c
// init/calibrate.c 这个函数会在main函数里面调用
void calibrate_delay(void)
{
	unsigned long lpj;
	static bool printed;
	int this_cpu = smp_processor_id();

	if (per_cpu(cpu_loops_per_jiffy, this_cpu)) {
		// 查看当前CPU的loops_per_jiffy是否已经被设置过了, 
		// 如果设置过了就直接使用这个值, 否则就进行校准.
		lpj = per_cpu(cpu_loops_per_jiffy, this_cpu);
		if (!printed)
			pr_info("Calibrating delay loop (skipped) "
				"already calibrated this CPU");
	} else if (preset_lpj) {
		// 如果在启动参数中指定了lpj的值, 就直接使用这个值, 否则就进行校准.
		lpj = preset_lpj;
		if (!printed)
			pr_info("Calibrating delay loop (skipped) "
				"preset value.. ");
	} else if ((!printed) && lpj_fine) {
		// 如果lpj_fine已经被设置了, 就直接使用这个值, 否则就进行校准.
		// 有些平台可由定时器频率直接给较准值
		lpj = lpj_fine;
		pr_info("Calibrating delay loop (skipped), "
			"value calculated using timer frequency.. ");
	} else if ((lpj = calibrate_delay_is_known())) {
		;
	} else if ((lpj = calibrate_delay_direct()) != 0) {
		// 尝试“直接计时法”进行校准, 如果成功了就使用这个值, 
		// 否则就进行“收敛法”校准.
		if (!printed)
			pr_info("Calibrating delay using timer "
				"specific routine.. ");
	} else {
		if (!printed)
			pr_info("Calibrating delay loop... ");
		lpj = calibrate_delay_converge();
	}
	per_cpu(cpu_loops_per_jiffy, this_cpu) = lpj;
	if (!printed)
		pr_cont("%lu.%02lu BogoMIPS (lpj=%lu)\n",
			lpj/(500000/HZ),
			(lpj/(5000/HZ)) % 100, lpj);

	loops_per_jiffy = lpj;
	printed = true;

	calibration_delay_done();
}
```

## 长延迟

内核中进行延迟的一个很直观的方法是比较当前的jiffies 和目标jiffies 

```c
unsigned long delay = jiffies + 100;
while (time_before(jiffies, delay));
// 延迟两秒
unsigned long delay = jiffies + 2*Hz;
while (time_before(jiffies, delay));
```

## 睡眠延迟

睡着延迟无疑是比忙等待更好的方式，睡着延迟在等待的时间到来之间进程处于睡眠状态，CPU 资源被其他进程使用

实际是创建一个定时器用于唤醒

```c
// timeout value in jiffies
signed long __sched schedule_timeout(signed long timeout)
{
	struct process_timer timer;
	unsigned long expire;

	switch (timeout)
	{
	case MAX_SCHEDULE_TIMEOUT:
		/*
		 * These two special cases are useful to be comfortable
		 * in the caller. Nothing more. We could take
		 * MAX_SCHEDULE_TIMEOUT from one of the negative value
		 * but I' d like to return a valid offset (>=0) to allow
		 * the caller to do everything it want with the retval.
		 */
		schedule();
		goto out;
	default:
		/*
		 * Another bit of PARANOID. Note that the retval will be
		 * 0 since no piece of kernel is supposed to do a check
		 * for a negative retval of schedule_timeout() (since it
		 * should never happens anyway). You just have the printk()
		 * that will tell you if something is gone wrong and where.
		 */
		if (timeout < 0) {
			printk(KERN_ERR "schedule_timeout: wrong timeout "
				"value %lx\n", timeout);
			dump_stack();
			current->state = TASK_RUNNING;
			goto out;
		}
	}
	// 计算超时时间
	expire = timeout + jiffies;

	timer.task = current;
	timer_setup_on_stack(&timer.timer, process_timeout, 0);
	// 创建定时器
	__mod_timer(&timer.timer, expire, MOD_TIMER_NOTPENDING);
	schedule();
	del_singleshot_timer_sync(&timer.timer);

	/* Remove the timer from the object tracker */
	destroy_timer_on_stack(&timer.timer);

	timeout = expire - jiffies;

 out:
	return timeout < 0 ? 0 : timeout;
}

static void process_timeout(struct timer_list *t)
{
	struct process_timer *timeout = from_timer(timeout, t, timer);

	wake_up_process(timeout->task);
}
```
