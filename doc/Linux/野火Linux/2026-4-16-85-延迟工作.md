# 延迟工作

`delayed_work` 是 Linux 内核中**基于工作队列（workqueue）+ 定时器（timer）** 的延迟任务执行机制，允许你将任务安排在**指定时间后**在**进程上下文**中执行，常用于需要**延迟处理**且**可睡眠**的场景（如设备驱动中的异步操作、周期性任务）。

## API

+   初始化

```c
// 静态初始化（推荐）
DECLARE_DELAYED_WORK(name, func);

// 动态初始化
INIT_DELAYED_WORK(struct delayed_work *dwork, work_func_t func);
INIT_DELAYED_WORK_ONSTACK(struct delayed_work *dwork, work_func_t func); // 栈上分配时使用
```

+   调度

```c
// 调度到默认工作队列，延迟 delay 个 jiffies 后执行
static inline bool schedule_delayed_work(struct delayed_work *dwork,
					 unsigned long delay);

// 调度到指定工作队列
static inline bool queue_delayed_work(struct workqueue_struct *wq,
				      struct delayed_work *dwork,
				      unsigned long delay);

// 常用时间转换宏
#define msecs_to_jiffies(ms)  // 毫秒转jiffies（推荐，兼容tickless内核）
#define HZ                    // 系统时钟频率（1秒的jiffies数）
```

+   取消

```c
// 尝试取消，不等待正在执行的任务
int cancel_delayed_work(struct delayed_work *dwork);

// 取消并等待任务完成（同步），确保任务不再执行
int cancel_delayed_work_sync(struct delayed_work *dwork);

// 刷新指定工作队列中的所有任务
void flush_workqueue(struct workqueue_struct *wq);
```

+   检查状态

```c
// 判断任务是否在队列中等待执行
bool delayed_work_pending(const struct delayed_work *dwork);
```

## 使用示例

```c
#include <linux/module.h>
#include <linux/workqueue.h>
#include <linux/jiffies.h>

// 定义延迟工作结构体和数据
struct my_data {
    struct delayed_work dwork;
    int count;
};

static struct my_data *data;

// 延迟工作回调函数（进程上下文，可睡眠）
static void my_delayed_work_handler(struct work_struct *work) {
    struct my_data *d = container_of(work, struct my_data, dwork.work);
    
    pr_info("Delayed work executed: count=%d, jiffies=%lu\n", 
           d->count++, jiffies);
    
    // 周期性执行（1秒后再次调度）
    schedule_delayed_work(&d->dwork, msecs_to_jiffies(1000));
}

static int __init my_init(void) {
    data = kzalloc(sizeof(*data), GFP_KERNEL);
    if (!data) return -ENOMEM;
    
    // 初始化延迟工作
    INIT_DELAYED_WORK(&data->dwork, my_delayed_work_handler);
    data->count = 0;
    
    // 首次调度：2秒后执行
    schedule_delayed_work(&data->dwork, msecs_to_jiffies(2000));
    
    pr_info("Module loaded, delayed work scheduled\n");
    return 0;
}

static void __exit my_exit(void) {
    // 取消延迟工作并同步等待完成
    cancel_delayed_work_sync(&data->dwork);
    kfree(data);
    pr_info("Module unloaded\n");
}

module_init(my_init);
module_exit(my_exit);
MODULE_LICENSE("GPL");
```
