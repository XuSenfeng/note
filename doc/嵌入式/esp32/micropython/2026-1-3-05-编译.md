# 编译

```bash
git clone https://github.com/micropython/micropython.git
cd micropython
make -C mpy-cross
cd ports/esp32
git submodule update --init --recursive

make clean
idf.py set-target esp32s3
IDF_TARGET=esp32s3 idf.py set-target esp32s3
IDF_TARGET=esp32s3 make BOARD=ESP32_GENERIC_S3 USER_C_MODULES=../../examples/usercmodule/micropython.cmake
# OTA变体: make BOARD=ESP32_GENERIC BOARD_VARIANT=OTA
picocom -b 115200 /dev/
```
