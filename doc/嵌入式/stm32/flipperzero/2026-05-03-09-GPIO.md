# GPIO

## 基础使用

```c
#include "furi_hal.h"

int gpio_control_main(void* p) {
    UNUSED(p);

    furi_hal_gpio_init_simple(&gpio_ext_pa7, GpioModeOutputPushPull);
    do {
        furi_hal_gpio_write(&gpio_ext_pa7, true);
        furi_delay_ms(500);
        furi_hal_gpio_write(&gpio_ext_pa7, false);
        furi_delay_ms(500);
    } while(furi_hal_gpio_read(&gpio_button_ok));
    furi_hal_gpio_init_simple(&gpio_ext_pa7, GpioModeAnalog);

    return 0;
}

```

## PWM

```bash
// 在PA7引脚设置一个10Hz的50占空比的PWM波形
furi_hal_pwm_start(FuriHalPwmOutputIdTim1PA7, 10, 50);
do {
    furi_delay_ms(500);
} while(furi_hal_gpio_read(&gpio_button_ok));
furi_hal_pwm_stop(FuriHalPwmOutputIdTim1PA7);
```
