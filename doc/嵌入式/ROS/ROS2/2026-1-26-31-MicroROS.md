# MicroROS

## 部署

### 电脑

需要运行一个docker的镜像监听端口连接开发板, 开启一个代理, 之后使用串口/蓝牙/以太网等形式连接开发板

在ros的src目录里面添加两个组件

```bash
git clone https://github.com/micro-ROS/micro-ROS-Agent.git -b $ROS_DISTRO
git clone https://github.com/micro-ROS/micro_ros_msgs.git -b $ROS_DISTRO
cd ..
colcon build
source install/setup.bash
ros2 run micro_ros_agent micro_ros_agent udp4 --port 8888
```

### 开发板

在platformio的配置文件添加使用的库以及通过wifi传输

```ini
board_microros_transport = wifi
lib_deps = 
    https://gitee.com/fishros/MPU6050_light.git
    https://gitee.com/fishros/micro_ros_platformio.git
```

初始化以及订阅消息

```cpp
#include <Arduino.h>
#include <WiFi.h>
#include <micro_ros_platformio.h>
#include <rcl/rcl.h>
#include <rclc/rclc.h>
#include <rclc/executor.h>
#include <geometry_msgs/msg/twist.h>

rcl_allocator_t allocator; // 动态内存分配器
rclc_support_t support; // 用于存储时钟, 内存分配器等支持结构
rclc_executor_t executor; // 执行器, 用于调度和执行ROS2回调函数
rcl_node_t node; // ROS2节点

rcl_subscription_t sub_cmd_vel; // 订阅者对象
geometry_msgs__msg__Twist msg_cmd_vel; // 存储接收到的cmd_vel消息

void twist_callback(const void * msgin) {
    geometry_msgs__msg__Twist * msg = (geometry_msgs__msg__Twist *)msgin;
    float linear_x = msg->linear.x;
    float angular_z = msg->angular.z;

    Serial.print("Received cmd_vel - Linear X: ");
    Serial.print(linear_x);
    Serial.print(", Angular Z: ");
    Serial.println(angular_z);

    // 在这里添加控制机器人运动的代码
}

// 创建一个任务运行micro-ROS
void micro_ros_task(void *arg) {
    // 1. 设置传输的协议以及等待完成
    Serial.println("WiFi connected, starting micro-ROS...");
    IPAddress agent_ip;
    agent_ip.fromString("192.168.31.101"); // 替换为micro-ROS代理的IP地址
    set_microros_wifi_transports("TP-LINK_5G_2B18", "13838106970", agent_ip, 8888);
    delay(2000);
    Serial.print("IP address: ");
    Serial.println(WiFi.localIP());

    //2. 初始化内存分配器和支持结构
    allocator = rcl_get_default_allocator();

    // 3. 初始化支持
    rclc_support_init(&support, 0, NULL, &allocator);
    // 4. 创建节点
    rclc_node_init_default(&node, "fishbot_motion_control", "", &support);
    // 5. 创建执行器, 可以处理2个回调函数句柄
    unsigned int number_handles = 2;    
    rclc_executor_init(&executor, &support.context, number_handles, &allocator);

    // 6. 创建订阅者
    rclc_subscription_init_best_effort(
        &sub_cmd_vel,
        &node,
        ROSIDL_GET_MSG_TYPE_SUPPORT(geometry_msgs, msg, Twist),
        "/cmd_vel"
    );
    // 7. 将订阅者添加到执行器
    rclc_executor_add_subscription(
        &executor,
        &sub_cmd_vel,
        &msg_cmd_vel,
        &twist_callback,
        ON_NEW_DATA
    );

    rclc_executor_spin(&executor); // 进入执行器自旋循环
}


void setup() { // put your setup code here, to run once:

    //初始化串口

    Serial.begin(115200);
    Serial.println("Serial initialized at 115200 baud");
    xTaskCreate(
        micro_ros_task,
        "micro_ros_task",
        10240,
        NULL,
        5,
        NULL
    );

}
void loop() { // put your main code here, to run repeatedly:

    Serial.println("My First PIO Project!");

    delay(1000);

}
```

+   发布消息

```cpp
rcl_publisher_t pub_odome; // 里程计发布者对象
nav_msgs__msg__Odometry msg_odome; // 发送的里程计消息
rcl_timer_t timer;    

// 8. 初始化里程计消息, 里面的字符串只是一个占位符, 需要根据需求动态分配
msg_odome.header.frame_id = micro_ros_string_utilities_set(msg_odome.header.frame_id, "odom");
msg_odome.child_frame_id = micro_ros_string_utilities_set(msg_odome.child_frame_id, "base_footprint");
// 9. 创建里程计发布者
rclc_publisher_init_best_effort(
    &pub_odome,
    &node,
    ROSIDL_GET_MSG_TYPE_SUPPORT(nav_msgs, msg, Odometry),
    "/odom"
);
// 10. 创建定时器, 每100ms触发一次
const unsigned int timer_timeout = 100; // milliseconds
rclc_timer_init_default(
    &timer,
    &support,
    RCL_MS_TO_NS(timer_timeout),
    timer_call_back
);
// 11. 将定时器添加到执行器
rclc_executor_add_timer(&executor, &timer);


// 时间同步
while(!rmw_uros_epoch_synchronized()) {
    Serial.println("Waiting for time synchronization...");
    rmw_uros_sync_session(1000);
    delay(100);
}
```
