# 网络服务

## 创建服务

### tcp服务

```go
package main

import "net"

func main() {
	// 监听本地8080端口
	ln, err := net.Listen("tcp", ":8080")
	if err != nil {
		panic(err)
	}
	defer ln.Close()

	for {
		// 接受连接
		conn, err := ln.Accept()
		if err != nil {
			continue
		}
		go handleConnection(conn)
	}
}

func handleConnection(conn net.Conn) {
	defer conn.Close()
	conn.Write([]byte("Hello, World!\n"))
}
```
