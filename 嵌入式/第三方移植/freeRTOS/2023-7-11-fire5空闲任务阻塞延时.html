    /* жү«жҸҸе°ұз»ӘеҲ—иЎЁдёӯжүҖжңүзәҝзЁӢзҡ„xTicksToDelayпјҢеҰӮжһңдёҚдёә0пјҢеҲҷеҮҸ1 */

        pxTCB = ( TCB_t * ) listGET_OWNER_OF_HEAD_ENTRY( ( &pxReadyTasksLists[i] ) );//иҝҷйҮҢеҸӘиҺ·еҸ–з¬¬дёҖдёӘйЎ№, жүҖд»Ҙй…ҚзҪ®дёәеҗҢдёҖдёӘдјҳе…Ҳзә§дёҚиғҪжӯЈеёёдҪҝз”Ё

    		pxTCB->xTicksToDelay --;
