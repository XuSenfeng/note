class MyWindow(QWidget):
    my_signal = pyqtSignal(str)

    def __init__(self):
        super().__init__()
        self.setupUI()
        self.msg_history = list()

    def setupUI(self):
        self.setGeometry(800, 200, 500, 200)
        self.setWindowTitle("Signal and Slot")

        container = QVBoxLayout()
        self.msg = QLabel("")
