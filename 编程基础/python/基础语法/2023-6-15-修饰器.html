import time

def timeit(repeats):
    def decorator(func):
        def wrapper(*args, **kwargs):
            total_time = 0
            for _ in range(repeats):		# 1000å†³å®šè¿™ä¸€ä¸ªå‡½æ•°çš„æ‰§è¡Œæ¬¡æ•°
                start_time = time.time()
                result = func(*args, **kwargs)	# è·å–ä¸€ä¸ªresult
                end_time = time.time()
                total_time += end_time - start_time
            average_time = total_time / repeats
            print(f"Average execution time: {average_time} seconds")
            return result
        return wrapper
    return decorator

    return x * 2
