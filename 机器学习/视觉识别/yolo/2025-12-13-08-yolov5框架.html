模型结构构建原理

在yolov5/models/yolov5*.yaml文件里面是对应的模型的结构

Backbone 提取多尺度特征 → Head 融合特征 → Detect 输出检测结果

# Ultralytics 🚀 AGPL-3.0 License - https://ultralytics.com/license

# Parameters
nc: 80 # number of classes 默认的分类数
# 实际的层的重复数量会乘这个数字
depth_multiple: 0.33 # model depth multiple 深度乘数，控制网络的深度（C3 模块的重复次数）。值越小，C3 重复次数越少
# 实际的输出通道数会乘以这个
width_multiple: 0.50 # layer channel multiple 宽度乘数，控制网络的通道数（每层卷积的输出通道数）。值越小，通道数越少，模型越窄
anchors: # 锚框 预定义的边界框尺寸，用于匹配不同大小的目标
  - [10, 13, 16, 30, 33, 23] # P3/8
  - [30, 61, 62, 45, 59, 119] # P4/16
  - [116, 90, 156, 198, 373, 326] # P5/32

# YOLOv5 v6.0 backbone
# 负责特征提取，基于 CSPNet 结构，通过卷积 + 残差模块提取多尺度语义特征
# from：输入来源，-1 表示上一层，数字表示对应层的索引（如 6 表示第 6 层）；
# number：该模块的重复次数；
# module：模块名称（Conv/C3/SPPF 等）；
# args：模块的参数（如卷积通道数、核大小等）
# 输出 P3/P4/P5 三个尺度的特征图（下采样 8/16/32 倍）
backbone:
  # [from, number, module, args]
  [
    # 0-P1/2 Conv 模块（卷积 + BN+SiLU 激活）：
    # 输出 64 通道，卷积核 6×6，步长 2，填充 2 → 下采样 2 倍（P1/2）。
    [-1, 1, Conv, [64, 6, 2, 2]], 
    [-1, 1, Conv, [128, 3, 2]], # 1-P2/4
    # C3 模块（CSPNet 核心，3 个卷积 + 残差块）：重复 3 次，通道 128 → 增强特征表达，无下采样
    [-1, 3, C3, [128]],
    [-1, 1, Conv, [256, 3, 2]], # 3-P3/8
    [-1, 6, C3, [256]],
    [-1, 1, Conv, [512, 3, 2]], # 5-P4/16
    [-1, 9, C3, [512]],
    [-1, 1, Conv, [1024, 3, 2]], # 7-P5/32
    [-1, 3, C3, [1024]],
    [-1, 1, SPPF, [1024, 5]], # 9
  ]

# YOLOv5 v6.0 head
# 实现多尺度特征融合，并通过 Detect 模块输出检测结果（边界框、置信度、类别）
# 先上采样融合浅层特征（细节），再下采样融合深层特征（语义），最终输出 3 个尺度的检测分支
head: [
  # 1×1 卷积降维：将 P5 特征从 1024 通道压缩到 512 → 减少计算量，为上采样做准备
    [-1, 1, Conv, [512, 1, 1]],
    # 上采样：最近邻插值，将特征图放大 2 倍 → 匹配 P4 尺度（16 倍下采样）
    [-1, 1, nn.Upsample, [None, 2, "nearest"]],
    # 特征拼接：将上采样后的 P5 特征（11 层）与 Backbone 的 P4 特征（6 层）按通道维度拼接
    #  → 融合深层语义 + 中层细节
    [[-1, 6], 1, Concat, [1]], # cat backbone P4
    # C3 模块：重复 3 次，通道 512，
    # False 表示无残差 → 融合后特征提纯（Head 部分的 C3 无残差，降低计算量）
    [-1, 3, C3, [512, False]], # 13

    [-1, 1, Conv, [256, 1, 1]],
    [-1, 1, nn.Upsample, [None, 2, "nearest"]],
    [[-1, 4], 1, Concat, [1]], # cat backbone P3
    [-1, 3, C3, [256, False]], # 17 (P3/8-small)

    [-1, 1, Conv, [256, 3, 2]],
    [[-1, 14], 1, Concat, [1]], # cat head P4
    [-1, 3, C3, [512, False]], # 20 (P4/16-medium)

    [-1, 1, Conv, [512, 3, 2]],
    [[-1, 10], 1, Concat, [1]], # cat head P5
    # C3 模块：P5/32 尺度检测分支（大目标），输出 1024 通道特征。
    [-1, 3, C3, [1024, False]], # 23 (P5/32-large)

    # Detect 模块：输入 17/20/23 层的 3 个尺度特征
    # 结合类别数 nc 和锚框 anchors，输出最终检测结果
    [[17, 20, 23], 1, Detect, [nc, anchors]], # Detect(P3, P4, P5)
  ]

class DetectionModel(BaseModel):
    """YOLOv5 detection model class for object detection tasks, supporting custom configurations and anchors."""

    def __init__(self, cfg="yolov5s.yaml", ch=3, nc=None, anchors=None):
        """Initializes YOLOv5 model with configuration file, input channels, number of classes, and custom anchors."""
        super().__init__()
        # 加载配置文件
        if isinstance(cfg, dict):
            self.yaml = cfg  # model dict
        else:  # is *.yaml
            import yaml  # for torch hub

            self.yaml_file = Path(cfg).name
            with open(cfg, encoding="ascii", errors="ignore") as f:
                self.yaml = yaml.safe_load(f)  # model dict
    		
        # Define model, 重新定义一下通道数
        ch = self.yaml["ch"] = self.yaml.get("ch", ch)  # input channels
        # 配置目标类别
        if nc and nc != self.yaml["nc"]:
            LOGGER.info(f"Overriding model.yaml nc={self.yaml['nc']} with nc={nc}")
            self.yaml["nc"] = nc  # override yaml value
        if anchors:
            LOGGER.info(f"Overriding model.yaml anchors with anchors={anchors}")
            self.yaml["anchors"] = round(anchors)  # override yaml value
        # 获取实际的模型
        self.model, self.save = parse_model(deepcopy(self.yaml), ch=[ch])  # model, savelist
        self.names = [str(i) for i in range(self.yaml["nc"])]  # default names
        self.inplace = self.yaml.get("inplace", True)

        # Build strides, anchors
        m = self.model[-1]  # Detect()
        if isinstance(m, (Detect, Segment)):

            def _forward(x):
                """Passes the input 'x' through the model and returns the processed output."""
                return self.forward(x)[0] if isinstance(m, Segment) else self.forward(x)

            s = 256  # 2x min stride
            m.inplace = self.inplace
            m.stride = torch.tensor([s / x.shape[-2] for x in _forward(torch.zeros(1, ch, s, s))])  # forward
            check_anchor_order(m)
            m.anchors /= m.stride.view(-1, 1, 1)
            self.stride = m.stride
            self._initialize_biases()  # only run once

        # Init weights, biases
        initialize_weights(self)
        self.info()
        LOGGER.info("")

def parse_model(d, ch):
    """Parses a YOLOv5 model from a dict `d`, configuring layers based on input channels `ch` and model architecture."""
    LOGGER.info(f"\n{'':>3}{'from':>18}{'n':>3}{'params':>10}  {'module':<40}{'arguments':<30}")
    # 获取模型的信息
    anchors, nc, gd, gw, act, ch_mul = (
        d["anchors"],
        d["nc"],
        d["depth_multiple"],
        d["width_multiple"],
        d.get("activation"),
        d.get("channel_multiple"),
    )
    # 激活函数
    if act:
        Conv.default_act = eval(act)  # redefine default activation, i.e. Conv.default_act = nn.SiLU()
        LOGGER.info(f"{colorstr('activation:')} {act}")  # print
    # 确保卷积层的输出通道数是 ch_mul 的整数倍，核心目的是适配硬件计算效率
    if not ch_mul:
        ch_mul = 8
    # 计算锚框的数量
    na = (len(anchors[0]) // 2) if isinstance(anchors, list) else anchors  # number of anchors
    # 计算输出的树立数量, +5指的是坐标以及置信度
    no = na * (nc + 5)  # number of outputs = anchors * (classes + 5)
    	
    # 层
    # 记录所有需要被后续层复用的层索引, 这些层的特征会被后续Concat/Detect层复用
    # 输出通道数
    layers, save, c2 = [], [], ch[-1]  # layers, savelist, ch out
    for i, (f, n, m, args) in enumerate(d["backbone"] + d["head"]):  # from, number, module, args
        # module进行实例化, 字符串=>实际的对象 
        m = eval(m) if isinstance(m, str) else m  # eval strings
        # 处理参数
        for j, a in enumerate(args):
            with contextlib.suppress(NameError):
                args[j] = eval(a) if isinstance(a, str) else a  # eval strings
    			# 实际的循环次数乘depth_multiple, 和1比较
        n = n_ = max(round(n * gd), 1) if n > 1 else n  # depth gain
        # 依据不同的模型构造参数
        if m in {
            Conv,
            GhostConv,
            Bottleneck,
            GhostBottleneck,
            SPP,
            SPPF,
            DWConv,
            MixConv2d,
            Focus,
            CrossConv,
            BottleneckCSP,
            C3,
            C3TR,
            C3SPP,
            C3Ghost,
            nn.ConvTranspose2d,
            DWConvTranspose2d,
            C3x,
        }:
          	# f是-1, 实际是上一层的输出, 第二个记录配置文件的输出通道
            c1, c2 = ch[f], args[0]
            # 乘上width_multiple同时保证可以整除
            if c2 != no:  # if not output
                c2 = make_divisible(c2 * gw, ch_mul)
    					
            args = [c1, c2, *args[1:]]
            # 以下几种的话需要插入参数n表示循环的次数
            if m in {BottleneckCSP, C3, C3TR, C3Ghost, C3x}:
                args.insert(2, n)  # number of repeats
                n = 1
        elif m is nn.BatchNorm2d:
            args = [ch[f]]
        elif m is Concat:
            c2 = sum(ch[x] for x in f)
        # TODO: channel, gw, gd
        elif m in {Detect, Segment}:
            args.append([ch[x] for x in f])
            if isinstance(args[1], int):  # number of anchors
                args[1] = [list(range(args[1] * 2))] * len(f)
            if m is Segment:
                args[3] = make_divisible(args[3] * gw, ch_mul)
        elif m is Contract:
            c2 = ch[f] * args[0] ** 2
        elif m is Expand:
            c2 = ch[f] // args[0] ** 2
        else:
            c2 = ch[f]
    			# 创建序列
        m_ = nn.Sequential(*(m(*args) for _ in range(n))) if n > 1 else m(*args)  # module
        t = str(m)[8:-2].replace("__main__.", "")  # module type
        np = sum(x.numel() for x in m_.parameters())  # number params
        m_.i, m_.f, m_.type, m_.np = i, f, t, np  # attach index, 'from' index, type, number params
        LOGGER.info(f"{i:>3}{f!s:>18}{n_:>3}{np:10.0f}  {t:<40}{args!s:<30}")  # print
        save.extend(x % i for x in ([f] if isinstance(f, int) else f) if x != -1)  # append to savelist
        # 记录出来
        layers.append(m_)
        if i == 0:
            ch = []
        ch.append(c2)
    return nn.Sequential(*layers), sorted(save)

class Conv(nn.Module):
    """Applies a convolution, batch normalization, and activation function to an input tensor in a neural network."""
    default_act = nn.SiLU()  # default activation
    	# 输入输出通道数, 卷积核, 步幅, 是不是paddings, groups, 是不是膨胀, 激活函数
    def __init__(self, c1, c2, k=1, s=1, p=None, g=1, d=1, act=True):
        """Initializes a standard convolution layer with optional batch normalization and activation."""
        super().__init__()
        self.conv = nn.Conv2d(c1, c2, k, s, autopad(k, p, d), groups=g, dilation=d, bias=False)
        self.bn = nn.BatchNorm2d(c2)
        self.act = self.default_act if act is True else act if isinstance(act, nn.Module) else nn.Identity()

    def forward(self, x):
        """Applies a convolution followed by batch normalization and an activation function to the input tensor `x`."""
        return self.act(self.bn(self.conv(x)))

    def forward_fuse(self, x):
        """Applies a fused convolution and activation function to the input tensor `x`."""
        return self.act(self.conv(x))
