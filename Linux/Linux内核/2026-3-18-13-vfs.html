VFS(Virtual Filesystem Switch)

2026-04-14
编辑本页

为用户提供统一的访问接口, 兼容不同的文件系统, 屏蔽底层细节

数据类型

进程
  |
  |  struct file（打开文件表）
  |    |
  |    |  f_path
  |    v
  |  struct path
  |    |------> struct vfsmount（挂载点）
  |    |------> struct dentry（目录项）
  |                    |
  |                    v
  |                struct inode（元数据）
  |
struct super_block（文件系统全局信息）
  • 超级块 (super_block)：表示一个已挂载的文件系统实例。Linux挂载文件系统时，会在内存创建一个超级块，用于管理文件系统。它包含了文件系统的元数据信息（如：文件系统类型、容量信息、结构参数、位置信息等），内核需要借助这些元数据信息才能访问文件系统
  • 索引节点 (inode)：表示文件系统中的一个文件或目录，用于访问文件系统中的文件。唯一标识一个文件, 记录文件的元数据以及提供了文件的操作接口，但是它并不包含的文件实际数据，因为inode属于抽象层，不能够和任何具体实现有耦合关系。每个inode都有一个唯一编号（inode号），用于在文件系统中索引文件
  • 目录项 (dentry)：表示文件路径中的一个分量，是Linux文件树的树节点，将文件路径名与对应的inode关联起来。每个文件和目录都会对应一个dentry，dentry之间通过父子关系形成一个树形结构（类似于家族树）每个dentry都是一个小的站点，文件路径就是导航路线，文件路径用于指导内核快速定位到目标文件对应的dentry
  • 文件对象 (file)：表示进程中一个已打开文件的实例，同一个文件可以被多个进程同时打开，每打开一次文件都要创建一个文件对象。

路径

struct path {
    struct vfsmount *mnt; // 指向挂载点的指针
    struct dentry *dentry; // 指向目录项的指针
} __randomize_layout;

挂载点

用于关联一个 super_block 和挂载根目录（dentry）。

  • 挂载点是指一个文件系统被“挂载”到现有目录树上的位置。
  • struct vfsmount 结构体描述了一个已挂载的文件系统实例，包括其超级块、挂载根目录等信息。
  • 每个挂载点都对应一个 struct vfsmount，内核通过它来区分不同的文件系统实例（如根文件系统、U盘、网络文件系统等）。
struct vfsmount {
  // 指向该挂载点的根目录项（dentry）。
  // 也就是这个文件系统在挂载后，作为根节点的目录项对象
    struct dentry *mnt_root;	/* root of the mounted tree */
  // 指向该挂载点对应的超级块（super_block）结构体。
  // 超级块保存了文件系统的全局信息（如类型、大小、操作函数等）
    struct super_block *mnt_sb;	/* pointer to superblock */
  // 挂载点的标志位，记录了该挂载点的各种属性（如只读、禁止执行、
  // 禁止设备文件、atime 行为等）
    int mnt_flags;

    ANDROID_KABI_RESERVE(1);
    ANDROID_KABI_RESERVE(2);
    ANDROID_KABI_RESERVE(3);
    ANDROID_KABI_RESERVE(4);
} __randomize_layout;

目录项

  • 目录项（directory entry，简称 dentry）是内核用来描述文件系统中每个路径分量（如 /home/user/file）节点的结构体。
  • struct dentry 结构体保存了文件名、父目录、指向 inode 的指针等信息。
  • dentry 主要用于路径查找缓存（dentry cache），加速文件系统操作，避免频繁访问磁盘
struct dentry {
    /* RCU lookup touched fields */
    unsigned int d_flags;		/* protected by d_lock */
    seqcount_spinlock_t d_seq;	/* per dentry seqlock */
    struct hlist_bl_node d_hash;	/* lookup hash list */
    struct dentry *d_parent;	/* 父目录 parent directory */
    struct qstr d_name; // 文件名
  // 指向使用的inode节点
    struct inode *d_inode;		/* Where the name belongs to - NULL is
    				 * negative */
    unsigned char d_iname[DNAME_INLINE_LEN];	/* small names */

    /* Ref lookup also touches following */
    struct lockref d_lockref;	/* per-dentry lock and refcount */
    const struct dentry_operations *d_op;
    struct super_block *d_sb;	/* 文件系统超级块 The root of the dentry tree */
    unsigned long d_time;		/* used by d_revalidate */
    void *d_fsdata;			/* fs-specific data */

    union {
    	struct list_head d_lru;		/* LRU list */
    	wait_queue_head_t *d_wait;	/* in-lookup ones only */
    };
    struct list_head d_child;	/* 插入父节点 child of parent list */
    struct list_head d_subdirs;	/* 子节点 our children */
    /*
     * d_alias and d_rcu can share memory
     */
    union {
    	struct hlist_node d_alias;	/* inode alias list */
    	struct hlist_bl_node d_in_lookup_hash;	/* only for in-lookup ones */

文件查找

需要先找到dentry, 因为dentry的d_inode和i_sb成员分别指向索引节点（inode）和超级块（super_block）。内核需要通过inode和super_block这两个对象来访问文件系统

内核会根据open函数传入的文件路径查找文件树，找到目标文件对应的dentry，再通过dentry找到inode。如果文件不存在（文件树中没有对应的dentry和inode），open函数可以创建文件并生成dentry和inode

获取到inode后，内核会创建一个新的file，同时将inode的关键信息复制给file，如：

  • file的f_op将指向inode的i_fop，file将直接使用inode的文件操作函数表。
  • file的f_mapping成员将指向inode的f_mmaping，通过file可以访问inode的页缓存。
  • 同时file的f_inode将指向inode，file和inode进行绑定。

内核最后会申请一个未使用的fd（文件描述符），将fd和file进行映射，并将这种映射关系记录在进程的已打开文件表。后续的文件操作，只需要通过fd查找进程已打开文件表获取file，就能通过file访问文件了，从而避免每次文件访问都需要索引Linux文件树

long do_sys_open(int dfd, const char __user *filename, int flags, umode_t mode)
{
    // 把传统 open/openat 的 flags、mode 规范化成 struct open_how 的小工具
    struct open_how how = build_open_how(flags, mode);
    return do_sys_openat2(dfd, filename, &how);
}

static long do_sys_openat2(int dfd, const char __user *filename,
    		   struct open_how *how)
{
    struct open_flags op;
    // 将的open_how结构体中的flags和mode字段转换为内核使用的open_flags结构体中的open_flag、
    // acc_mode和lookup_flags字段。它还会检查用户传入的flags和mode是否合法，并根据需要设置一些隐含的标志。
    int fd = build_open_flags(how, &op);
    struct filename *tmp;

    if (fd)
    	return fd; // 出错了，直接返回错误码

    // 把用户空间的filename参数转换为内核空间的filename结构体。
    // getname函数会检查用户传入的指针是否合法，并且会把字符串复制到内核空间中。如果出错了，返回一个错误码。
    tmp = getname(filename);
    if (IS_ERR(tmp))
    	return PTR_ERR(tmp);

    // 获取一个新的文件描述符，这个描述符是未使用的，并且会设置一些标志位（如O_CLOEXEC）。
    // 如果没有可用的文件描述符了，返回一个错误码。
    fd = get_unused_fd_flags(how->flags);
    if (fd >= 0) {
    	// 处理文件的打开逻辑，调用do_filp_open函数来打开文件，并返回一个file结构体指针。
    	struct file *f = do_filp_open(dfd, tmp, &op);
    	if (IS_ERR(f)) {
    		put_unused_fd(fd);
    		fd = PTR_ERR(f);
    	} else {
    		// 如果文件成功打开了，就把这个文件结构体指针和文件描述符关联起来，
    		// 并且通知内核文件系统监视器（fsnotify）这个文件被打开了。
    		fsnotify_open(f);
    		fd_install(fd, f);
    	}
    }
    // 无论成功还是失败，都要释放之前分配的内核空间的filename结构体。
    putname(tmp);
    return fd;
}


struct file *do_filp_open(int dfd, struct filename *pathname,
    	const struct open_flags *op)
{
    struct nameidata nd;
    int flags = op->lookup_flags;
    struct file *filp;

    // 建立本次查找上下文 nameidata（起始目录 fd、路径名、状态栈等）
    set_nameidata(&nd, dfd, pathname);
    // 先尝试 RCU 模式，失败后再尝试正常模式，最后尝试重新验证模式
    filp = path_openat(&nd, op, flags | LOOKUP_RCU);
    if (unlikely(filp == ERR_PTR(-ECHILD)))
    	// 如果 RCU 模式失败，说明路径可能被修改了，尝试正常模式
    	filp = path_openat(&nd, op, flags);
    // 如果正常模式也失败了，说明路径可能被删除了，尝试重新验证模式
    if (unlikely(filp == ERR_PTR(-ESTALE)))
    	filp = path_openat(&nd, op, flags | LOOKUP_REVAL);
    restore_nameidata();
    return filp;
}

路径查找

路径解析是通过路径遍历将路径名字符串映射到对应dentry的过程，是 open ()、stat () 等系统调用的必备环节

实际实现的逻辑是: 从根目录 / 当前工作目录的已知 dentry 出发，逐级匹配路径组件的子 dentry，完成全路径遍历

一、RCU 快速查找（LOOKUP_RCU）

定位：读多写少场景下的最快路径，无锁、无引用计数、无写开销，基于 RCU 与 seqcount 实现。

核心原理

  • 全程持有 rcu_read_lock()，保证 dentry/inode 等关键结构不被释放（内存不回收），但允许修改。
  • 不获取 dentry 引用计数、不碰 d_lock 自旋锁，通过 seqcount 序列号检测并发修改。
  • 仅在 dentry 缓存（dcache）中查找，不访问底层文件系统。

执行流程

  1. 进入 RCU 读临界区（rcu_read_lock()）。
  2. 调用 lookup_fast()，通过 __d_lookup_rcu() 在 dcache 哈希表中无锁查找目标组件。
  3. 用 read_seqcount_retry() 校验序列号，确保访问的 dentry/inode 未被修改。
  4. 成功则返回；失败 / 不满足条件（如需睡眠、需锁）则回退到常规查找。

二、常规查找（REF-walk）

定位：RCU 快速查找失败后的稳定 fallback，基于引用计数与自旋锁，保证强一致性。

核心原理

  • 放弃 RCU 模式，转为引用计数 + 锁的并发控制，确保数据稳定。

  • 核心机制：

    • dentry->d_lockref：自旋锁 + 引用计数，原子操作保护 dentry 稳定性。
    • d_lock：防止 dentry 重命名 / 解除链接，保证父目录与名称不变。
    • rename_lock/mount_lock：seqlock 检测重命名 / 挂载变化，失败则重试。
    • inode->i_rwsem：读写信号量，序列化目录修改。

执行流程

  1. RCU 查找失败后，切换到 REF-walk 模式。
  2. 对目标 dentry 增加引用计数（dget()），确保其不被释放。
  3. 按需获取 d_lock/i_rwsem 等锁，校验 dentry 有效性。
  4. 访问底层文件系统（如需要），完成最终查找。
  5. 若返回 -ESTALE（陈旧缓存），则回退到强制重校验查找。

三、强制重校验查找（LOOKUP_REVAL）

定位：常规查找返回陈旧（-ESTALE）时的终极校验，强制刷新缓存，确保绝对准确。

核心原理

  • 仅在常规查找发现缓存条目过旧 / 不可信时触发。
  • 核心动作：调用文件系统的 ->d_revalidate() 方法，强制重新验证 dentry 缓存信息。
  • 不使用 RCU，完全依赖 REF-walk 的锁与引用计数机制。

执行流程

  1. 常规查找返回 -ESTALE（如 NFS 服务器重启、文件被删除后重建）。
  2. 调用者设置 LOOKUP_REVAL 标志，重新执行查找。
  3. 对所有相关 dentry 执行 强制重校验，确认缓存与底层文件系统一致。
  4. 校验成功则返回；失败则返回错误码。

数组越界处理

static inline unsigned long array_index_mask_nospec(unsigned long index,
   					    unsigned long size)
{
   /*
    * Always calculate and emit the mask even if the compiler
    * thinks the mask is not needed. The compiler does not take
    * into account the value of @index under speculation.
    */
   OPTIMIZER_HIDE_VAR(index);
   return ~(long)(index | (size - 1UL - index)) >> (BITS_PER_LONG - 1);
}
  • 若 $ index<size $，则 $size−1−index≥0$，t 的最高位通常是 0。
  • 若$ index≥size$，size-1-index 在无符号下会下溢成一个很大的数，最高位变 1，于是 t 最高位也会是 1。
  1. 获取是否越界

转成有符号并取反(long)t：

  • 最高位 0 -> 正数
  • 最高位 1 -> 负数

再 ~(long)t 后：

  • 原来正数 -> 取反后最高位 1
  • 原来负数 -> 取反后最高位 0
  1. 取一个掩码

右移到只看符号位扩展：

  • 最高位 1 -> 结果全 1（即 ~0UL）
  • 最高位 0 -> 结果全 0

所以最终得到：

  • index < size -> mask = ~0UL
  • index >= size -> mask = 0

ssize_t vfs_read(struct file *file, char __user *buf, size_t count, loff_t *pos)
{
    ssize_t ret;
    
    // 检查一下权限
    if (!(file->f_mode & FMODE_READ))
    	return -EBADF; // 文件描述符没有以可读方式打开
    if (!(file->f_mode & FMODE_CAN_READ))
    	return -EINVAL; // 文件描述符不支持读取操作
    if (unlikely(!access_ok(buf, count)))
    	return -EFAULT; // 用户空间缓冲区不可访问

    // 验证读取区域是否合法，包括检查偏移量和长度是否合理，
    // 以及是否存在强制锁定等情况。如果验证失败，返回相应的错误码。
    ret = rw_verify_area(READ, file, pos, count);
    if (ret)
    	return ret;
    if (count > MAX_RW_COUNT)
    	count =  MAX_RW_COUNT;
    // 根据文件操作结构体中的函数指针，调用相应的读取函数来执行读取操作。
    // 如果文件操作结构体中没有定义 ->read 和 ->read_iter 函数，那么这是不被支持的情况，
    // 会返回 -EINVAL 错误码。
    if (file->f_op->read)
    	ret = file->f_op->read(file, buf, count, pos);
    else if (file->f_op->read_iter)
    	ret = new_sync_read(file, buf, count, pos);
    else
    	ret = -EINVAL;
    if (ret > 0) {
    	// 读取成功后，调用 fsnotify_access() 来通知文件系统发生了访问事件，
    	// 并且调用 add_rchar() 来更新当前进程的读字符数统计。
    	fsnotify_access(file);
    	add_rchar(current, ret);
    }
    // 来增加当前进程的系统调用读次数统计，并返回读取的字节数或者错误码。
    inc_syscr(current);
    return ret;
}
