AIO

2026-04-18
编辑本页

AIO (Asynchronous I/O，异步 I/O) 是 Linux 系统中一种非阻塞的 I/O 处理模型，允许应用程序发起 I/O 操作后立即返回，继续执行其他任务，而无需等待 I/O 操作完成。当内核完成 I/O 操作 (包括数据准备和从内核空间到用户空间的数据拷贝两个阶段) 后，会主动通知应用程序。

支持三种通知方式：信号通知、线程回调、无通知 (需主动查询)

可以使用信号通知和线程通知两种两种方式

对比项 旧代码（信号通知 SIGEV_SIGNAL） 新代码（线程通知 SIGEV_THREAD）
通知方式 依赖 Linux 信号机制 依赖 线程机制（无任何信号）
处理函数类型 信号处理函数（受严格限制） 普通线程回调函数（无限制）
代码复杂度 高（需配置 sigaction、信号掩码、标志位） 低（仅配置 sigevent，无信号代码）
函数参数格式 固定：int, siginfo_t*, void* 标准：sigval_t
函数调用限制 只能用信号安全函数（不能随便用 printf） 无限制（可调用任何 C 库函数）
执行环境 中断主进程执行（信号抢占） 独立线程执行（不干扰主进程）
可靠性 信号可能丢失、重入 线程回调稳定可靠

API

异步读写处理函数

// 请求对一个有效的文件描述符进行异步读操作
int aio_read(struct aiocb *aiocbp);
// 用来请求一个异步写操作
int aio_write(struct aiocb *aiocbp);

获取执行状态以及是否报错

// 用来确定请求的状态
// EINPROGRESS：说明请求尚未完成。ECANCELLED：说明请求被应用程序取消了。
// -1：说明发生了错误，具体错误原因由errno 记录
int aio_error( struct aiocb *aiocbp );

获取实际的执行返回值

ssize_t aio_return( struct aiocb *aiocbp );

阻塞调用进程, 直到异步请求结束

int aio_suspend( const struct aiocb *const cblist[], int n, const struct timespec *timeout );

取消对某个文件描述符执行的一个或所有I/O 请求

要取消一个请求，用户需提供文件描述符和aiocb 指针。如果这个请求被成功取消了，那么这个函数就会返回AIO_CANCELED。如果请求完成了，这个函数就会返回 AIO_NOTCANCELED

int aio_cancel( int fd, struct aiocb *aiocbp );

批量处理大量IO操作

// mode 参数可以是LIO_WAIT 或LIO_NOWAIT。LIO_WAIT 会阻塞这个调用，直到所有的 I/O
// 都完成为止。在操作进行排队之后，LIO_NOWAIT 就会返回。list 是一个aiocb 引用的列表，最大
// 元素的个数是由nent 定义的。如果list 的元素为NULL，lio_listio()会将其忽略
int lio_listio( int mode, struct aiocb *list[], int nent, struct sigevent *sig );

示例

不使用回调

int fd, ret;
struct aiocb my_aiocb;  // AIO控制块（栈变量，不是指针）

fd = open("file.txt", O_RDONLY);  // 打开文件
if (fd < 0) perror("open");

bzero(&my_aiocb, sizeof(my_aiocb));  // 【关键】清零整个结构体（含aio_sigevent）

my_aiocb.aio_buf = malloc(BUFSIZE + 1);  // 分配读缓冲区
if (!my_aiocb.aio_buf) perror("malloc");

// 配置核心读参数
my_aiocb.aio_fildes = fd;     // 文件描述符
my_aiocb.aio_nbytes = BUFSIZE;// 读多少字节
my_aiocb.aio_offset = 0;      // 从文件开头读

ret = aio_read(&my_aiocb);     // 发起异步读，立刻返回
if (ret < 0) perror("aio_read");

// 【轮询核心】不停查询I/O是否完成
while (aio_error(&my_aiocb) == EINPROGRESS)
    continue;  // I/O进行中就空转

// 【修正笔误】my_iocb → my_aiocb
if ((ret = aio_return(&my_aiocb)) > 0) {
    // 读取成功，ret=实际读到的字节数
} else {
    // 读取失败
}

线程处理回调

#include <aio.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <fcntl.h>
#include <errno.h>

#define BUF_SIZE 1024

// 线程回调函数（AIO完成后自动执行）
void aio_completion_handler(sigval_t sigval) {
    struct aiocb *req = (struct aiocb *)sigval.sival_ptr;

    // 1. 先检查I/O是否成功完成
    if (aio_error(req) == 0) {
        ssize_t ret = aio_return(req);  // 获取读取的字节数
        printf("AIO读取完成，字节数：%zd\n", ret);
    }

    // 2. 释放内存（仅在这里释放，主函数不要重复释放）
    free(req->aio_buf);
    free(req);
}

int main() {
    int fd = open("test.txt", O_RDONLY);
    if (fd == -1) { perror("open"); exit(1); }

    // 分配AIO控制块
    struct aiocb *req = malloc(sizeof(struct aiocb));
    memset(req, 0, sizeof(struct aiocb));

    // 初始化AIO参数
    req->aio_buf = malloc(BUF_SIZE);
    req->aio_fildes = fd;
    req->aio_nbytes = BUF_SIZE;
    req->aio_offset = 0;

    // 线程通知配置
    struct sigevent sev;
    memset(&sev, 0, sizeof(sev));
    sev.sigev_notify = SIGEV_THREAD;        // 线程通知
    sev.sigev_notify_function = aio_completion_handler;
    sev.sigev_notify_attributes = NULL;
    sev.sigev_value.sival_ptr = req;        // 传递AIO块指针
    req->aio_sigevent = sev; // 进行复制赋值

    // 提交异步读请求
    if (aio_read(req) == -1) { perror("aio_read"); exit(1); }
    printf("AIO请求已提交，主程序继续执行...\n");

    // 修正BUG：req是指针，直接传req，不是&req
    // 空循环等待I/O完成（简单用法）
    while (aio_error(req) == EINPROGRESS) {
        usleep(1000); // 优化：加休眠，避免CPU 100%占用
    }

    close(fd);
    return 0;
}

ret = aio_read(&my_aiocb);

处理函数

// 信号处理函数（固定参数格式：SA_SIGINFO 模式要求）
void aio_completion_handler(int signo, siginfo_t *info, void *context)
{
    struct aiocb *req;  // 指向对应的异步I/O控制块

    // 第37行：校验信号是否是我们绑定的SIGIO
    if (info->si_signo == SIGIO) {
        // 第38行：从信号附加信息中，拿到之前的AIO控制块指针
        req = (struct aiocb*)info->si_value.sival_ptr;

        // 第41行：检查异步I/O是否**无错误完成**
        if (aio_error(req) == 0) {
            // 第43行：获取I/O结果（对应同步read的返回值：实际读取的字节数）
            ret = aio_return(req);
        }
    }
}

驱动实现

每个I/O 请求都对应于一个kiocb 结构体，其ki_filp 成员指向对应的file 指针, 通过is_sync_kiocb()可以判断某kiocb 是否为同步I/O 请求，如果返回非真，表示为异步I/O 请求

块设备和网络设备本身是异步的，只有字符设备必须明确表明应支持AIO

read_iter/write_iter 是 Linux 内核统一的 I/O 入口接口，它自动兼容「同步读写」「异步 AIO」「io_uring」三种场景，驱动里直接写同步逻辑就够了, 你在函数里返回 -EIOCBQUEUED → 它才变成异步 I/O, 只有你的设备本身不支持立即完成 I/O时，才需要异步

  1. read_iter/write_iter 是通用 I/O 接口，不是异步专用
  2. 同步实现是默认、标准、推荐写法
  3. 异步只是可选功能，不需要就别写
// 内核异步 I/O 的标准读写接口(替代老旧的 read/write), 必须实现的函数
// 参数 struct kiocb = 内核版的 aiocb（和你用户态的 struct aiocb 一一对应）
// 参数 struct iov_iter = 内核管理用户态数据缓冲区的迭代器
ssize_t (*read_iter) (struct kiocb *, struct iov_iter *);
ssize_t (*write_iter) (struct kiocb *, struct iov_iter *);
// 高性能异步 I/O 轮询函数, 专门用于 NVMe SSD 等高速设备；
// 不使用硬件中断，而是CPU 自旋轮询 I/O 完成，降低延迟
int (*iopoll)(struct kiocb *kiocb, bool spin);

同步处理示例

#include <linux/module.h>
#include <linux/fs.h>
#include <linux/cdev.h>
#include <linux/uaccess.h>
#include <linux/ioctl.h>

/* 设备参数 */
#define DEV_NAME    "demo_iter"
#define BUF_SIZE    1024

/* 全局设备结构体 */
struct demo_dev {
    struct cdev cdev;
    char buf[BUF_SIZE];  // 内核态数据缓冲区
    size_t len;          // 有效数据长度
};

static struct demo_dev *demo_devp;
static dev_t dev_no;

// ------------------------------
// 核心：现代 read_iter 实现
// ------------------------------
static ssize_t demo_read_iter(struct kiocb *iocb, struct iov_iter *iter)
{
    struct file *file = iocb->ki_filp;
    struct demo_dev *dev = file->private_data;
    loff_t *ppos = &iocb->ki_pos;
    size_t count;
    int ret;

    /* 偏移检查 */
    if (*ppos >= dev->len)
        return 0;

    /* 最大读取长度限制 */
    count = min(iov_iter_count(iter), (size_t)(dev->len - *ppos));

    /* 内核缓冲区 → 用户空间（iov_iter 自动处理所有用户态拷贝） */
    ret = copy_to_iter(dev->buf + *ppos, count, iter);
    *ppos += ret;

    return ret;
}

// ------------------------------
// 核心：现代 write_iter 实现
// ------------------------------
static ssize_t demo_write_iter(struct kiocb *iocb, struct iov_iter *iter)
{
    struct file *file = iocb->ki_filp;
    struct demo_dev *dev = file->private_data;
    loff_t *ppos = &iocb->ki_pos;
    size_t count;
    int ret;

    /* 偏移和缓冲区溢出检查 */
    if (*ppos >= BUF_SIZE)
        return -ENOSPC;

    count = min(iov_iter_count(iter), (size_t)(BUF_SIZE - *ppos));

    /* 用户空间 → 内核缓冲区 */
    ret = copy_from_iter(dev->buf + *ppos, count, iter);
    *ppos += ret;
    if (ret > dev->len)
        dev->len = ret;

    return ret;
}

/* 文件操作集合：仅实现 read_iter/write_iter 即可 */
static const struct file_operations demo_fops = {
    .owner      = THIS_MODULE,
    .read_iter  = demo_read_iter,    // 现代读接口
    .write_iter = demo_write_iter,   // 现代写接口
};

/* 模块初始化 */
static int __init demo_init(void)
{
    /* 申请设备号 + 注册字符设备 */
    alloc_chrdev_region(&dev_no, 0, 1, DEV_NAME);
    demo_devp = kzalloc(sizeof(*demo_devp), GFP_KERNEL);

    cdev_init(&demo_devp->cdev, &demo_fops);
    cdev_add(&demo_devp->cdev, dev_no, 1);
    printk(KERN_INFO "demo_iter: 加载成功，主设备号 %d\n", MAJOR(dev_no));
    return 0;
}

static void __exit demo_exit(void)
{
    cdev_del(&demo_devp->cdev);
    kfree(demo_devp);
    unregister_chrdev_region(dev_no, 1);
}

module_init(demo_init);
module_exit(demo_exit);
MODULE_LICENSE("GPL");

}
