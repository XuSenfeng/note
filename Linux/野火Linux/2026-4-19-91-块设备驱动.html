块设备驱动

2026-04-26
编辑本页

字符设备与块设备I/O操作的不同如下。

  1. 块设备只能以块为单位接受输入和返回输出，而字符设备则以字节为单位。大多数设备 是字符设备，因为它们不需要缓冲而且不以固定块大小进行操作。
  2. 块设备对于I/O 请求有对应的缓冲区，因此它们可以选择以什么顺序进行响应，字符设 备无须缓冲且被直接读写。对于存储设备而言调整读写的顺序作用巨大，因为在读写连续的扇区 比分离的扇区更快。
  3. 字符设备只能被顺序读写，而块设备可以随机访问。虽然块设备可随机访问，但是对于 磁盘这类机械设备而言，顺序地组织块设备的访问可以提高性能，对扇区1、10、 3、2的请求被调整为对扇区1、2、3、10的请求。而对SD卡、RamDisk等块设备而言，不存在 机械上的原因，进行这样的调整没有必要

操作接口函数

struct block_device_operations {
    // 提交一个块设备 I/O 请求的核心函数。旁路掉传统的请求队列
    //（Request Queue），直接将 bio（块 I/O 请求结构）传递给驱动进行
    // 处理。对于无队列（Make Request 或 blk-mq 直通）的块设备非常关键
    blk_qc_t (*submit_bio) (struct bio *bio);
    // 当通过设备节点打开块设备时被调用（例如执行 mount 或者使用 
    // dd 直接读写设备时）。用于增加设备的引用计数、执行初始化、检查介质状态等
    int (*open) (struct block_device *, fmode_t);
    // 在最后一次关闭该块设备节点时被调用（例如 umount 或者关闭设备文件）
    void (*release) (struct gendisk *, fmode_t);
    //用于直接对指定的物理页（page）进行块级别的直接读写操作，
    // 而不经过内存分配较重的 bio 层。通常用于 Swap 子系统（休眠或内核交换区）
    // 直接和高速存储（如 NVMe、ZRAM）互发数据。被调用时上下文环境较为严苛
    int (*rw_page)(struct block_device *, sector_t, struct page *, unsigned int);
    // 处理发送到该块设备的特定 I/O 控制命令。块设备层无法处理的 
    // ioctl 指令最后会被下发到这里，驱动可以借此实现私有的芯片控制逻辑
    // 大部分的标准请求内核处理
    int (*ioctl) (struct block_device *, fmode_t, unsigned, unsigned long);
    // 运行在 64 位内核上的 32 位用户空间程序提供 ioctl 兼容性层转换
    int (*compat_ioctl) (struct block_device *, fmode_t, unsigned, unsigned long);
    // 一般用于可移动介质（比如 CD-ROM、SD 读卡器），
    // 定期检查并报告是否有介质变动（比如插入或拔出了存储卡/光盘）
    unsigned int (*check_events) (struct gendisk *disk,
    			      unsigned int clearing);
    // 解锁设备的隐藏或原生容量。有些硬盘在 BIOS 层面或默认状态下会隐藏最高容量
    void (*unlock_native_capacity) (struct gendisk *);
    // 当介质改变时（如刚插入新的 U 盘或 SD 卡），用于驱动层重新验证介
    // 质信息并更新磁盘的拓扑结构和大小状态
    int (*revalidate_disk) (struct gendisk *);
    // 获取传统的物理硬件几何结构（Geometry，即柱面 C、磁头 H、扇区 S 
    // 的信息 CHS）。这通常是为了向前兼容早期的 BIOS 和老式硬盘工具（如 fdisk）
    int (*getgeo)(struct block_device *, struct hd_geometry *);
    /* this callback is with swap_lock and sometimes page table lock held */
    void (*swap_slot_free_notify) (struct block_device *, unsigned long);
    int (*report_zones)(struct gendisk *, sector_t sector,
    		unsigned int nr_zones, report_zones_cb cb, void *data);
    char *(*devnode)(struct gendisk *disk, umode_t *mode);
    struct module *owner;
    const struct pr_ops *pr_ops;
};

gendisk结构体

使用gendisk（通用磁盘）结构体来表示一个独立的磁盘设备（或分区）

它负责把底层的硬件设备抽象出来，并管理着该设备的物理参数、分区信息、请求队列以及设备驱动的操作接口等

struct gendisk {
    /* major, first_minor and minors are input parameters only,
     * don't use directly.  Use disk_devt() and disk_max_parts().
     */
    // 块设备使用的主设备号（Major Number）。主设备号一般与特定的驱动程序相关联
    int major;			/* major number of driver */
    // 该磁盘分配的第一个次设备号
    int first_minor;
    // 该磁盘支持的最大次设备号数量。它直接决定了这个设备能有多少个分区
    int minors;                     /* maximum number of minors, =1 for
                                         * disks that can't be partitioned. */
    // 字符数组，存储设备在系统中的名称，例如 "sda", "nvme0n1", "mmcblk0" 等
    char disk_name[DISK_NAME_LEN];	/* name of major driver */
    
    // 支持的事件掩码，比如是否支持介质改变（光盘更换）或安全弹出事件
    unsigned short events;		/* supported events */
    // 事件处理相关的标志，指示怎么处理这些事件
    unsigned short event_flags;	/* flags related to event processing */

    /* Array of pointers to partitions indexed by partno.
     * Protected with matching bdev lock but stat and other
     * non-critical accesses use RCU.  Always access through
     * helpers.
     */
    // 指向 struct disk_part_tbl（分区表）的指针。在这个表里包含了一个
    // 指针数组，存储了当前磁盘下面所有的逻辑分区
    struct disk_part_tbl __rcu *part_tbl;
    // 代表整个磁盘设备本身（即分区 0）。在 Linux 中，整块磁盘的概念也按照一
    // 个“特殊的分区”来统一管理，这样读写全盘和读写分区的流程可以高度复用
    struct hd_struct part0;
    
    // 处理操作的函数
    const struct block_device_operations *fops;
    struct request_queue *queue; // 指向该块设备的请求队列
    void *private_data; // 提供给底层的具体驱动程序保留其私有的设备管理信息

    int flags;
    unsigned long state;
#define GD_NEED_PART_SCAN		0
    struct rw_semaphore lookup_sem;
    struct kobject *slave_dir;

    struct timer_rand_state *random;
    atomic_t sync_io;		/* RAID */
    struct disk_events *ev;
#ifdef  CONFIG_BLK_DEV_INTEGRITY
    struct kobject integrity_kobj;
#endif	/* CONFIG_BLK_DEV_INTEGRITY */
#if IS_ENABLED(CONFIG_CDROM)
    struct cdrom_device_info *cdi;
#endif
    int node_id;
    // 指向“坏块”（Bad blocks）管理表，由于一些设备（如 pmem/nvdimm 内存盘）
    // 无法自己屏蔽坏扇区，需要块设备层用软件来记录和避让
    struct badblocks *bb;
    struct lockdep_map lockdep_map;
};

请求

在Linux块设备驱动中，使用request结构体来表征等待进行的I/O请求

在 Linux 的存储 I/O 栈中，文件系统下发的基本单位是 bio（表示一次连续逻辑地址的独立 I/O）。而 struct request 是块设备层面的“加工产物”。块层会把多个物理上相邻的 bio 聚合并合并（Merge）成一个个更大的 request，随后将其放入 I/O 调度器进行排序，最终分发给底层的硬件驱动（如 NVMe、SCSI 驱动）去执行

struct request {
    // 该请求所属的全局请求队列
    struct request_queue *q;
    // 指向软件队列（Software Queue）上下文
    // 通常每个 CPU 核心有一个以避免锁竞争
    struct blk_mq_ctx *mq_ctx;
    // 指向硬件派发队列（Hardware Dispatch Context）。
    // 映射到底层存储控制器实际的硬件队列机制
    struct blk_mq_hw_ctx *mq_hctx;

    unsigned int cmd_flags;		/* op and common flags */
    req_flags_t rq_flags;

    int tag;
    int internal_tag;

    /* the following two fields are internal, NEVER access directly */
    unsigned int __data_len;	/* total data len */
    // 该请求要访问的起始物理扇区
    sector_t __sector;		/* sector cursor */

    struct bio *bio;
    struct bio *biotail;

    struct list_head queuelist;
    

    // 为了优化 Cacheline，生命周期不在同一阶段的变量被放在了 union 里组合复用
    /*
     * The hash is used inside the scheduler, and killed once the
     * request reaches the dispatch list. The ipi_list is only used
     * to queue the request for softirq completion, which is long
     * after the request has been unhashed (and even removed from
     * the dispatch list).
     */
    union {
    	struct hlist_node hash;	/* merge hash */
    	struct list_head ipi_list;
    };

    /*
     * The rb_node is only used inside the io scheduler, requests
     * are pruned when moved to the dispatch queue. So let the
     * completion_data share space with the rb_node.
     */
    union {
    	struct rb_node rb_node;	/* sort/lookup */
    	struct bio_vec special_vec;
    	void *completion_data;
    	int error_count; /* for legacy drivers, don't use */
    };

    /*
     * Three pointers are available for the IO schedulers, if they need
     * more they have to dynamically allocate it.  Flush requests are
     * never put on the IO scheduler. So let the flush fields share
     * space with the elevator data.
     */
    union {
    	struct {
    		struct io_cq		*icq;
    		void			*priv[2];
    	} elv;

    	struct {
    		unsigned int		seq;
    		struct list_head	list;
    		rq_end_io_fn		*saved_end_io;
    	} flush;
    };

    struct gendisk *rq_disk;
    struct hd_struct *part;
#ifdef CONFIG_BLK_RQ_ALLOC_TIME
    /* Time that the first bio started allocating this request. */
    u64 alloc_time_ns;
#endif
    /* Time that this request was allocated for this IO. */
    u64 start_time_ns;
    /* Time that I/O was submitted to the device. */
    u64 io_start_time_ns;

#ifdef CONFIG_BLK_WBT
    unsigned short wbt_flags;
#endif
    /*
     * rq sectors used for blk stats. It has the same value
     * with blk_rq_sectors(rq), except that it never be zeroed
     * by completion.
     */
    unsigned short stats_sectors;

    /*
     * Number of scatter-gather DMA addr+len pairs after
     * physical address coalescing is performed.
     */
    unsigned short nr_phys_segments;

#if defined(CONFIG_BLK_DEV_INTEGRITY)
    unsigned short nr_integrity_segments;
#endif

#ifdef CONFIG_BLK_INLINE_ENCRYPTION
    struct bio_crypt_ctx *crypt_ctx;
    struct blk_ksm_keyslot *crypt_keyslot;
#endif

    unsigned short write_hint;
    unsigned short ioprio;

    enum mq_rq_state state;
    refcount_t ref;

    unsigned int timeout;
    unsigned long deadline;

    union {
    	struct __call_single_data csd;
    	u64 fifo_time;
    };

    /*
     * completion callback.
     */
    rq_end_io_fn *end_io;
    void *end_io_data;

    ANDROID_KABI_RESERVE(1);
};

调度算法

Linux 5.10 已全面采用 blk-mq（Multi-Queue Block IO Queueing Mechanism） 多队列框架，替代了传统单队列设计，支持的调度器均为多队列优化版本。核心调度器包括：mq-deadline（默认）、kyber、bfq、none（noop）

blk-mq 是 Linux 4.10+ 引入的核心架构，解决传统单队列锁竞争问题，适配多核 CPU 与高速存储设备。

  • 核心设计：为每个 CPU 核心维护独立请求队列，减少跨核同步开销
  • 请求流程：提交 → 软件队列（per-CPU）→ 硬件队列（设备）→ 执行
  • 关键优势：多核扩展性强、降低锁竞争、提升 NVMe/SSD 等高速设备性能
调度器 核心目标 延迟控制 吞吐量 公平性 CPU 开销 最佳适用设备
mq-deadline 低延迟 + 吞吐量平衡 ★★★★☆ ★★★★☆ ★★★☆☆ 低 机械硬盘、混合负载
kyber 低延迟优先 ★★★★★ ★★★★☆ ★★★☆☆ 中 NVMe/SSD、数据库
bfq 公平带宽 + 低延迟 ★★★★☆ ★★★★☆ ★★★★★ 中高 桌面、多媒体、多租户
none 极简 + 设备自主 ★★☆☆☆ ★★★★★ ★★☆☆☆ 极低 NVMe/SSD、RAM 盘
